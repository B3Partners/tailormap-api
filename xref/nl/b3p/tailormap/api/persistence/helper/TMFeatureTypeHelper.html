7   package nl.b3p.tailormap.api.persistence.helper;
8   
9   import java.util.Optional;
10  import nl.b3p.tailormap.api.persistence.Application;
11  import nl.b3p.tailormap.api.persistence.TMFeatureType;
12  import nl.b3p.tailormap.api.persistence.json.AppLayerSettings;
13  import nl.b3p.tailormap.api.persistence.json.AppTreeLayerNode;
14  
15  public class TMFeatureTypeHelper {
16    public static boolean isEditable(
17        Application application, AppTreeLayerNode appTreeLayerNode, TMFeatureType featureType) {
18      if (featureType == null) {
19        return false;
20      }
21  
22      boolean editable = false;
23      // Currently FeatureSourceHelper#loadCapabilities() sets editable to true when the datastore is
24      // a JDBC DataStore (even when the database account can't write to the table, can't detect this
25      // without trying). Other DataStore types we support (only WFS atm) we don't set as writeable
26  
27      // TODO: in the future, check for authorizations on editing. Currently you only need to be
28      // logged in (the viewer frontend already checks this before showing editable layers so we don't
29      // need to check for an authenticated user here).
30      if (featureType.isWriteable()) {
31        AppLayerSettings appLayerSettings = application.getAppLayerSettings(appTreeLayerNode);
32        editable =
33            Optional.ofNullable(appLayerSettings).map(AppLayerSettings::getEditable).orElse(false);
34      }
35      return editable;
36    }
37  }
