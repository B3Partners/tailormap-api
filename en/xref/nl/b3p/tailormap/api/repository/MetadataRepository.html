6   package nl.b3p.tailormap.api.repository;
7   
8   import nl.tailormap.viewer.config.metadata.Metadata;
9   
10  import org.springframework.data.jpa.repository.JpaRepository;
11  import org.springframework.transaction.annotation.Transactional;
12  
13  /**
14   * Easy to use methods to access {@link Metadata}.
15   *
16   * @since 0.1
17   */
18  public interface MetadataRepository extends JpaRepository<Metadata, Long> {
19  
20      /**
21       * Find a metadata value using a known key such as {@link Metadata#DATABASE_VERSION_KEY} or
22       * {@link Metadata#DEFAULT_APPLICATION}.
23       *
24       * @param configKey known key
25       * @return the found value or {@code null}
26       */
27      Metadata findByConfigKey(String configKey);
28  
29      /**
30       * Delete a value from the metadata using a known key such as {@link
31       * Metadata#DEFAULT_APPLICATION}.
32       *
33       * @param configKey known key
34       */
35      @Transactional
36      void deleteMetadataByConfigKey(String configKey);
37  }
