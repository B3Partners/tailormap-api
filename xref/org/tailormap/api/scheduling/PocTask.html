6   package org.tailormap.api.scheduling;
7   
8   import java.lang.invoke.MethodHandles;
9   import java.time.Instant;
10  import org.quartz.DisallowConcurrentExecution;
11  import org.quartz.JobDataMap;
12  import org.quartz.JobExecutionContext;
13  import org.quartz.PersistJobDataAfterExecution;
14  import org.slf4j.Logger;
15  import org.slf4j.LoggerFactory;
16  import org.springframework.lang.NonNull;
17  import org.springframework.scheduling.quartz.QuartzJobBean;
18  
19  /** POC task for testing purposes. This will only log messages and keep a counter. */
20  @DisallowConcurrentExecution
21  @PersistJobDataAfterExecution
22  public class PocTask extends QuartzJobBean {
23    private static final Logger logger =
24        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
25  
26    @Override
27    protected void executeInternal(@NonNull JobExecutionContext context) {
28      // NOTE: This immutable map is a snapshot of the job data maps at the time of the job execution.
29      JobDataMap mergedJobDataMap = context.getMergedJobDataMap();
30      // NOTE: This map is mutable and can be used to store job data.
31      JobDataMap jobDataMap = context.getJobDetail().getJobDataMap();
32  
33      int i = 1 + (int) mergedJobDataMap.getOrDefault("executions", 0);
34      jobDataMap.put("executions", i);
35      jobDataMap.put("lastExecutionFinished", Instant.now());
36      context.setResult("POC task executed successfully");
37  
38      logger.info(
39          "POC task {}:{} executing, details follow:",
40          context.getJobDetail().getKey().getGroup(),
41          context.getJobDetail().getKey().getName());
42      mergedJobDataMap.forEach((key, value) -> logger.info("   {}: {}", key, value));
43    }
44  }
