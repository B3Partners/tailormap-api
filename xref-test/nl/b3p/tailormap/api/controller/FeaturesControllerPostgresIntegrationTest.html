387     // page 1
388     MvcResult result =
389         mockMvc
390             .perform(get(provinciesWFS).param("page", "1"))
391             .andExpect(status().isOk())
392             .andExpect(content().contentType(MediaType.APPLICATION_JSON))
393             .andExpect(jsonPath("$.total").value(exactWfsCounts ? provinciesWFSTotalCount : -1))
394             .andExpect(jsonPath("$.page").value(1))
395             .andExpect(jsonPath("$.pageSize").value(pageSize))
396             .andExpect(jsonPath("$.features").isArray())
397             .andExpect(jsonPath("$.features").isNotEmpty())
398             .andExpect(jsonPath("$.features[0]").isMap())
399             .andExpect(jsonPath("$.features[0]").isNotEmpty())
400             .andExpect(jsonPath("$.features[0].__fid").isNotEmpty())
401             .andExpect(jsonPath("$.features[0].geometry").isEmpty())
402             .andExpect(jsonPath("$.features[0].attributes.naam").value("Drenthe"))
403             .andExpect(jsonPath("$.features[0].attributes.ligtInLandNaam").value("Nederland"))
404             .andExpect(jsonPath("$.columnMetadata").isArray())
405             .andExpect(jsonPath("$.columnMetadata").isNotEmpty())
406             .andReturn();
407 
408     String body = result.getResponse().getContentAsString();
409     logger.trace(body);
410     assertNotNull(body, "response body should not be null");
411     List<Service> page1Features = JsonPath.read(body, "$.features");
412     assertEquals(
413         pageSize,
414         page1Features.size(),
415         () ->
416             "there should be "
417                 + pageSize
418                 + " provinces in the list, but was "
419                 + page1Features.size());
420 
421     // page 2
422     result =
423         // bestturlijke gebieden WFS; provincies
424         mockMvc
425             .perform(get(provinciesWFS).param("page", "2"))
426             .andExpect(status().isOk())
427             .andExpect(content().contentType(MediaType.APPLICATION_JSON))
428             .andExpect(jsonPath("$.total").value(exactWfsCounts ? provinciesWFSTotalCount : -1))
429             .andExpect(jsonPath("$.page").value(2))
430             .andExpect(jsonPath("$.pageSize").value(pageSize))
431             .andExpect(jsonPath("$.features").isArray())
432             .andExpect(jsonPath("$.features").isNotEmpty())
433             .andExpect(jsonPath("$.columnMetadata").isArray())
434             .andExpect(jsonPath("$.columnMetadata").isNotEmpty())
435             .andReturn();
436 
437     body = result.getResponse().getContentAsString();
438     logger.trace(body);
439     assertNotNull(body, "response body should not be null");
440 
441     List<Service> page2Features = JsonPath.read(body, "$.features");
442     assertEquals(
443         2,
444         page2Features.size(),
445         () -> "there should be 2 provinces in the list, but was " + page2Features.size());
446 
447     // check for duplicates
448     int retrievedFeatsCount = page1Features.size() + page2Features.size();
449     page2Features.addAll(page1Features);
450     assertEquals(
451         retrievedFeatsCount,
452         (int) page2Features.stream().distinct().count(),
453         "there should be no duplicates in 2 sequential pages");
454   }
