7   
8   import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
9   import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
10  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.jsonPath;
11  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
12  
13  import nl.b3p.tailormap.api.HSQLDBTestProfileJPAConfiguration;
14  import nl.b3p.tailormap.api.repository.ApplicationRepository;
15  import nl.b3p.tailormap.api.security.SecurityConfig;
16  
17  import org.junit.jupiter.api.MethodOrderer;
18  import org.junit.jupiter.api.Order;
19  import org.junit.jupiter.api.Test;
20  import org.junit.jupiter.api.TestMethodOrder;
21  import org.springframework.beans.factory.annotation.Autowired;
22  import org.springframework.boot.autoconfigure.EnableAutoConfiguration;
23  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
24  import org.springframework.boot.test.context.SpringBootTest;
25  import org.springframework.http.MediaType;
26  import org.springframework.test.context.ActiveProfiles;
27  import org.springframework.test.web.servlet.MockMvc;
28  
29  @SpringBootTest(
30          classes = {
31              HSQLDBTestProfileJPAConfiguration.class,
32              FeaturesController.class,
33              SecurityConfig.class
34          })
35  @AutoConfigureMockMvc
36  @EnableAutoConfiguration
37  @ActiveProfiles("test")
38  @TestMethodOrder(MethodOrderer.OrderAnnotation.class)
39  class FeaturesControllerIntegrationTest {
40      @Autowired private MockMvc mockMvc;
41      @Autowired ApplicationRepository applicationRepository;
42  
43      @Test
44      @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
45      void filter_not_supported() throws Exception {
46          mockMvc.perform(get("/app/1/layer/2/features").param("filter", "naam=Utrecht"))
47                  .andExpect(status().is4xxClientError())
48                  .andExpect(content().contentType(MediaType.APPLICATION_JSON))
49                  .andExpect(jsonPath("$.code").value(400));
50      }
51  
52      @Test
53      @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
54      void fid_not_supported() throws Exception {
55          mockMvc.perform(
56                          get("/app/1/layer/2/features")
57                                  .param(
58                                          "__fid",
59                                          "Provinciegebied.19ce551e-bc01-46e9-b953-929318dcdf87"))
60                  .andExpect(status().is4xxClientError())
61                  .andExpect(content().contentType(MediaType.APPLICATION_JSON))
62                  .andExpect(jsonPath("$.code").value(400));
63      }
64  
65      @Test
66      @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
67      void givenOnly_XorY_shouldError() throws Exception {
68          mockMvc.perform(get("/app/1/layer/2/features").param("x", "3"))
69                  .andExpect(status().is4xxClientError())
70                  .andExpect(content().contentType(MediaType.APPLICATION_JSON))
71                  .andExpect(jsonPath("$.code").value(400));
72  
73          mockMvc.perform(get("/app/1/layer/2/features").param("y", "3"))
74                  .andExpect(status().is4xxClientError())
75                  .andExpect(content().contentType(MediaType.APPLICATION_JSON))
76                  .andExpect(jsonPath("$.code").value(400));
77      }
78  
79      @Test
80      @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
81      void given_distance_NotGreaterThanZero() throws Exception {
82          mockMvc.perform(get("/app/1/layer/2/features?y=3&y=3").param("distance", "0"))
83                  .andExpect(status().is4xxClientError())
84                  .andExpect(content().contentType(MediaType.APPLICATION_JSON))
85                  .andExpect(jsonPath("$.code").value(400));
86  
87          mockMvc.perform(get("/app/1/layer/2/features?y=3&y=3").param("distance", "-1"))
88                  .andExpect(status().is4xxClientError())
89                  .andExpect(content().contentType(MediaType.APPLICATION_JSON))
90                  .andExpect(jsonPath("$.code").value(400));
91      }
92  
93      @Test
94      @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
95      void should_error_when_calling_with_nonexistent_appId() throws Exception {
96          mockMvc.perform(get("/app/400/layer/1/features"))
97                  .andExpect(status().isNotFound())
98                  .andExpect(content().contentType(MediaType.APPLICATION_JSON))
99                  .andExpect(jsonPath("$.code").value(404));
100     }
101 
102     @Test
103     @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
104     void should_not_find_when_called_without_appId() throws Exception {
105         mockMvc.perform(get("/app/layer/features")).andExpect(status().isNotFound());
106     }
107 
108     @Test
109     /* this test changes database content */
110     @Order(Integer.MAX_VALUE)
111     @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
112     void should_send_401_when_application_login_required() throws Exception {
113         applicationRepository.setAuthenticatedRequired(1L, true);
114 
115         mockMvc.perform(get("/app/1/layer/2/features").accept(MediaType.APPLICATION_JSON))
116                 .andExpect(status().isUnauthorized())
117                 .andExpect(content().contentType(MediaType.APPLICATION_JSON))
118                 .andExpect(jsonPath("$.code").value(401))
119                 .andExpect(jsonPath("$.url").value("/login"));
120     }
121 }
