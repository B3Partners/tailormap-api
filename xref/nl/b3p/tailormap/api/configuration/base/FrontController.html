6   package nl.b3p.tailormap.api.configuration.base;
7   
8   import org.springframework.stereotype.Controller;
9   import org.springframework.web.bind.annotation.GetMapping;
10  
11  /**
12   * Front controller which forwards requests to URLs created by the Angular routing module to the
13   * proper viewer or admin index.html of the Angular bundle, for example when the user refreshes the
14   * page. Unfortunately, a proper front controller would first check if the file which the requested
15   * URL maps to exists and only return the index.html file if it doesn't. That is quite complicated
16   * to do in Spring Boot so as a poor-mans alternative we explicitly match the specific router URLs
17   * navigated to by the Angular frontends which don't map to files from the bundle.
18   *
19   * <p>It could happen that a frontend developer adds a route and forgets to update this front
20   * controller, and it is only noticed later when someone presses F5 and gets a 404.
21   */
22  @Controller
23  public class FrontController {
24    @GetMapping(value = {"/login", "/app/**", "/service/**"})
25    public String appIndex() {
26      return "forward:/index.html";
27    }
28  
29    @GetMapping(
30        value = {
31          "/admin",
32          "/admin/login",
33          "/admin/catalog",
34          "/admin/catalog/**", // Extra line needed in order to match slashes. Regexps won't match.
35          "/admin/user**",
36          "/admin/group**",
37          "/admin/applications",
38          "/admin/applications/**"
39        })
40    public String adminIndex() {
41      return "forward:/admin/index.html";
42    }
43  
44    @GetMapping(value = {"/swagger-ui", "/swagger-ui/"})
45    public String swaggerUiWelcomePage() {
46      return "redirect:/swagger-ui/index.html";
47    }
48  }
