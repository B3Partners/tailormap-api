415     MvcResult result =
416         mockMvc
417             .perform(

420                     .with(setServletPath(url))
421                     .param("page", "1"))
422             .andExpect(status().isOk())
423             .andExpect(content().contentType(MediaType.APPLICATION_JSON))
424             .andExpect(jsonPath("$.total").value(exactWfsCounts ? provinciesWFSTotalCount : -1))
425             .andExpect(jsonPath("$.page").value(1))
426             .andExpect(jsonPath("$.pageSize").value(pageSize))
427             .andExpect(jsonPath("$.features").isArray())
428             .andExpect(jsonPath("$.features").isNotEmpty())
429             .andExpect(jsonPath("$.features[0]").isMap())
430             .andExpect(jsonPath("$.features[0]").isNotEmpty())
431             .andExpect(jsonPath("$.features[0].__fid").isNotEmpty())
432             .andExpect(jsonPath("$.features[0].geometry").isEmpty())
433             .andExpect(jsonPath("$.features[0].attributes.naam").value("Zuid-Holland"))
434             .andExpect(jsonPath("$.features[0].attributes.ligtInLandNaam").value("Nederland"))
435             .andExpect(jsonPath("$.columnMetadata").isArray())
436             .andExpect(jsonPath("$.columnMetadata").isNotEmpty())
437             .andReturn();
438 
439     String body = result.getResponse().getContentAsString();
440     logger.trace(body);
441     assertNotNull(body, "response body should not be null");
442     List<Service> page1Features = JsonPath.read(body, "$.features");
443     assertEquals(
444         pageSize,
445         page1Features.size(),
446         () ->
447             "there should be "
448                 + pageSize
449                 + " provinces in the list, but was "
450                 + page1Features.size());
451 
452     // page 2
453     result =
454         // bestuurlijke gebieden WFS; provincies
455         mockMvc
456             .perform(
457                 get(url)
458                     .accept(MediaType.APPLICATION_JSON)
459                     .with(setServletPath(url))
460                     .param("page", "2"))
461             .andExpect(status().isOk())
462             .andExpect(content().contentType(MediaType.APPLICATION_JSON))
463             .andExpect(jsonPath("$.total").value(exactWfsCounts ? provinciesWFSTotalCount : -1))
464             .andExpect(jsonPath("$.page").value(2))
465             .andExpect(jsonPath("$.pageSize").value(pageSize))
466             .andExpect(jsonPath("$.features").isArray())
467             .andExpect(jsonPath("$.features").isNotEmpty())
468             .andExpect(jsonPath("$.columnMetadata").isArray())
469             .andExpect(jsonPath("$.columnMetadata").isNotEmpty())
470             .andReturn();
