46  @RestController
47  @Validated
48  @RequestMapping(path = "/map/{appid}", produces = MediaType.APPLICATION_JSON_VALUE)
49  public class MapController {
50      private final Log logger = LogFactory.getLog(getClass());
51      @Autowired private ApplicationRepository applicationRepository;
52  
53      @Operation(
54              summary = "",
55              tags = {},
56              responses = {
57                  @ApiResponse(
58                          responseCode = "200",
59                          description = "OK",
60                          content =
61                                  @Content(
62                                          mediaType = MediaType.APPLICATION_JSON_VALUE,
63                                          schema = @Schema(implementation = MapResponse.class))),
64                  @ApiResponse(
65                          responseCode = "400",
66                          description = "Bad Request",
67                          content =
68                                  @Content(
69                                          mediaType = MediaType.APPLICATION_JSON_VALUE,
70                                          schema = @Schema(implementation = ErrorResponse.class)))
71              })
72      @GetMapping(produces = MediaType.APPLICATION_JSON_VALUE)
73      @ResponseBody
74      public MapResponse get(
75              @Parameter(name = "appid", description = "application id", required = true)
76                      @PathVariable("appid")
77                      Long appid) {
78          logger.trace("Requesting 'map' for application id: " + appid);
79  
80          MapResponse mapResponse = new MapResponse();
81  
82          // this could throw EntityNotFound, which is handles by handleEntityNotFoundException
83          // and in a normal flow this should not happen
84          // as appid is (should be) validated by calling the /app/ endpoint
85          Application application = applicationRepository.getById(appid);
86  
87          getApplicationParams(application, mapResponse);
88          getLayers(application, mapResponse);
89  
90          return mapResponse;
91      }
92  
93      @ExceptionHandler(EntityNotFoundException.class)
94      @ResponseStatus(
95              value =
96                      HttpStatus
97                              .BAD_REQUEST /*,reason = "Bad Request" -- adding 'reason' will drop the body */)
98      @ResponseBody
99      public ErrorResponse handleEntityNotFoundException(EntityNotFoundException exception) {
100         logger.warn(
101                 "Requested an application that does not exist. Message: " + exception.getMessage());
102         return new ErrorResponse()
103                 .message("Requested an application that does not exist")
104                 .code(400);
105     }
106 
107     private void getApplicationParams(@NotNull Application a, @NotNull MapResponse mapResponse) {
108         final String pCode = a.getProjectionCode();
109         //        final String pCode =
110         //                "EPSG:28992[+proj=sterea +lat_0=52.15616055555555 +lon_0=5.38763888888889
111         // +k=0.9999079 +x_0=155000 +y_0=463000 +ellps=bessel
112         // +towgs84=565.237,50.0087,465.658,-0.406857,0.350733,-1.87035,4.0812 +units=m +no_defs]";
113         CoordinateReferenceSystem c = new CoordinateReferenceSystem();
114         if (null != pCode) {
115             c.code(pCode.substring(0, pCode.indexOf('[')))
116                     .definition(pCode.substring(pCode.indexOf('[') + 1, pCode.lastIndexOf(']')));
117         }
118         Bounds maxExtent = new Bounds();
119         if (null != a.getMaxExtent()) {
120             maxExtent
121                     .minx(a.getMaxExtent().getMinx())
122                     .miny(a.getMaxExtent().getMiny())
123                     .maxx(a.getMaxExtent().getMaxx())
124                     .maxy(a.getMaxExtent().getMaxy())
125                     .crs(a.getMaxExtent().getCrs().getName());
126         }
127         Bounds initialExtent = new Bounds();
128         if (null != a.getStartExtent()) {
129             initialExtent
130                     .minx(a.getStartExtent().getMinx())
131                     .miny(a.getStartExtent().getMiny())
132                     .maxx(a.getStartExtent().getMaxx())
133                     .maxy(a.getStartExtent().getMaxy())
134                     .crs(a.getStartExtent().getCrs().getName());
135         }
136 
137         mapResponse.crs(c).maxExtent(maxExtent).initialExtent(initialExtent);
138     }
139 
140     private void getLayers(@NotNull Application a, @NotNull MapResponse mapResponse) {
141         List<StartLayer> startLayers = a.getStartLayers();
142         if (null != a.getStartLayers()) {
143             for (StartLayer l : startLayers) {
144                 AppLayer appLayer =
145                         new AppLayer()
146                                 .id(l.getApplicationLayer().getId())
147                                 .displayName(l.getApplicationLayer().getLayerName())
148                                 .url(l.getApplicationLayer().getService().getUrl())
149                                 .serviceId(l.getApplicationLayer().getService().getId())
150                                 // TODO fixup hardcoded data
151                                 .crs(new CoordinateReferenceSystem())
152                                 .visible(true)
153                                 .isBaseLayer(true);
154 
155                 GeoService geoService = l.getApplicationLayer().getService();
156                 Service s =
157                         new Service()
158                                 .url(geoService.getUrl())
159                                 .id(geoService.getId())
160                                 .name(geoService.getName())
161                                 .protocol(Service.ProtocolEnum.fromValue(geoService.getProtocol()));
162                 if (geoService.getProtocol().equalsIgnoreCase(TileService.PROTOCOL)) {
163                     s.tilingProtocol(
164                             Service.TilingProtocolEnum.fromValue(
165                                     ((TileService) geoService).getTilingProtocol()));
166 
167                     // TODO
168                     // find first matrix set that sort of matches on the numeric part of the CRS
169                     // code
170                     TileMatrixSet tileMatrixSet =
171                             ((TileService) geoService)
172                                     .getMatrixSets().stream()
173                                             .filter(
174                                                     tms ->
175                                                             tms.getCrs()
176                                                                     .contains(
177                                                                             appLayer.getCrs()
178                                                                                     .getCode()
179                                                                                     .substring(5)))
180                                             .findFirst()
181                                             .orElse(null);
182                     if (null != tileMatrixSet) {
183                         // you cannot add the JSONObject or the tileMatrixSet; that just does not
184                         // work,
185                         // but you can add a Map
186                         s.matrixSet(tileMatrixSet.toJSONObject().toMap());
187                     }
188                 }
189                 if (appLayer.getIsBaseLayer()) {
190                     mapResponse.addBaseLayersItem(appLayer);
191                 }
192                 mapResponse.addServicesItem(s);
193             }
194         }
195     }
