23  public class WFSFeatureSourceHelper implements FeatureSourceHelper {
24      private static final Log log = LogFactory.getLog(WFSFeatureSourceHelper.class);
25  
26      public static DataStore createDataStore(
27              Map<String, Object> extraDataStoreParams, WFSFeatureSource fs) throws IOException {
28          Map<String, Object> params = new HashMap<>();
29  
30          // Params which can be overridden
31          params.put(WFSDataStoreFactory.TIMEOUT.key, TIMEOUT);
32  
33          if (extraDataStoreParams != null) {
34              params.putAll(extraDataStoreParams);
35          }
36  
37          // Params which can not be overridden below
38          String wfsUrl = fs.getUrl();
39          if (!wfsUrl.endsWith("&") && !wfsUrl.endsWith("?")) {
40              wfsUrl += wfsUrl.contains("?") ? "&" : "?";
41          }
42          wfsUrl = wfsUrl + "REQUEST=GetCapabilities&SERVICE=WFS";
43          if (!wfsUrl.toUpperCase().contains("VERSION")) {
44              wfsUrl += "&VERSION=1.1.0";
45          }
46  
47          params.put(WFSDataStoreFactory.URL.key, wfsUrl);
48          params.put(WFSDataStoreFactory.USERNAME.key, fs.getUsername());
49          params.put(WFSDataStoreFactory.PASSWORD.key, fs.getPassword());
50  
51          Map<String, Object> logParams = new HashMap<>(params);
52          if (fs.getPassword() != null) {
53              logParams.put(
54                      WFSDataStoreFactory.PASSWORD.key,
55                      String.valueOf(new char[fs.getPassword().length()]).replace("\0", "*"));
56          }
57          log.debug("Opening datastore using parameters: " + logParams);
58          DataStore ds = DataStoreFinder.getDataStore(params);
59          if (ds == null) {
60              throw new IOException("Cannot open datastore using parameters " + logParams);
61          }
62          return ds;
63      }
64  
65      public static SimpleFeatureSource openGeoToolsFSFeatureSource(
66              WFSFeatureSource fs, SimpleFeatureType sft, int timeout) throws IOException {
67          Map<String, Object> extraParams = new HashMap<>();
68          extraParams.put(WFSDataStoreFactory.TIMEOUT.key, timeout);
69          DataStore ds = WFSFeatureSourceHelper.createDataStore(extraParams, fs);
70  
71          return ds.getFeatureSource(sft.getTypeName());
72      }
73  
74      @Override
75      public SimpleFeatureSource openGeoToolsFeatureSource(FeatureSource fs, SimpleFeatureType sft)
76              throws IOException {
77          return WFSFeatureSourceHelper.openGeoToolsFSFeatureSource(
78                  (WFSFeatureSource) fs, sft, TIMEOUT);
79      }
80  
81      @Override
82      public SimpleFeatureSource openGeoToolsFeatureSource(
83              FeatureSource fs, SimpleFeatureType sft, int timeout) throws IOException {
84          return WFSFeatureSourceHelper.openGeoToolsFSFeatureSource(
85                  (WFSFeatureSource) fs, sft, timeout);
86      }
87  }
