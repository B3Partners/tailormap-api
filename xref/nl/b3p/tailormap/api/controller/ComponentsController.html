40  import javax.persistence.EntityNotFoundException;
41  
42  @RestController
43  @Validated
44  @RequestMapping(path = "/app/{appId}/components", produces = MediaType.APPLICATION_JSON_VALUE)
45  public class ComponentsController {
46  
47      private final Log logger = LogFactory.getLog(getClass());
48      @Autowired private ApplicationRepository applicationRepository;
49  
50      /**
51       * Handle any {@code EntityNotFoundException} that this controller might throw while getting the
52       * application.
53       *
54       * @param exception the exception
55       * @return an error response
56       */
57      @ExceptionHandler(EntityNotFoundException.class)
58      @ResponseStatus(
59              value =
60                      HttpStatus
61                              .NOT_FOUND /*,reason = "Not Found" -- adding 'reason' will drop the body */)
62      @ResponseBody
63      public ErrorResponse handleEntityNotFoundException(EntityNotFoundException exception) {
64          logger.warn(
65                  "Requested an application that does not exist. Message: " + exception.getMessage());
66          return new ErrorResponse()
67                  .message("Requested an application that does not exist")
68                  .code(HttpStatus.NOT_FOUND.value());
69      }
70  
71      /**
72       * GET /components/{appId}
73       *
74       * @param appId application id (required)
75       * @return OK (status code 200)
76       */
77      @Operation(
78              summary = "",
79              tags = {},
80              responses = {
81                  @ApiResponse(
82                          responseCode = "200",
83                          description = "OK",
84                          content =
85                                  @Content(
86                                          mediaType = MediaType.APPLICATION_JSON_VALUE,
87                                          schema = @Schema(implementation = Component.class))),
88                  @ApiResponse(
89                          responseCode = "400",
90                          description = "Bad Request",
91                          content =
92                                  @Content(
93                                          mediaType = MediaType.APPLICATION_JSON_VALUE,
94                                          schema = @Schema(implementation = ErrorResponse.class))),
95                  @ApiResponse(
96                          responseCode = "401",
97                          description = "Unauthorized",
98                          content =
99                                  @Content(
100                                         mediaType = MediaType.APPLICATION_JSON_VALUE,
101                                         schema = @Schema(implementation = RedirectResponse.class)))
102             })
103     @GetMapping(produces = MediaType.APPLICATION_JSON_VALUE)
104     public ResponseEntity<?> get(
105             @Parameter(name = "appId", description = "application id", required = true)
106                     @PathVariable("appId")
107                     Long appId) {
108         logger.trace("Requesting 'components' for application id: " + appId);
109 
110         // this could throw EntityNotFound, which is handled by #handleEntityNotFoundException
111         // and in a normal flow this should not happen
112         // as appId is (should be) validated by calling the /app/ endpoint
113         Application application = applicationRepository.getById(appId);
114         if (application.isAuthenticatedRequired() && !AuthUtil.isAuthenticatedUser()) {
115             // login required, send RedirectResponse
116             return ResponseEntity.status(HttpStatus.UNAUTHORIZED).body(new RedirectResponse());
117         } else {
118             List<Component> components = new ArrayList<>();
119             findComponents(application, components);
120             return ResponseEntity.status(HttpStatus.OK).body(components);
121         }
122     }
123 
124     /**
125      * find all configured components for this application.
126      *
127      * @param application the application at hand
128      * @param componentList the list that holds the components
129      */
130     private void findComponents(Application application, List<Component> componentList) {
131         logger.trace("listing components: " + application.getComponents());
132         componentList.addAll(
133                 application.getComponents().stream()
134                         .map(
135                                 p ->
136                                         new Component()
137                                                 .putConfigItem("title", p.getName())
138                                                 .type(p.getClassName())
139                                                 .config(p.getDetails()))
140                         .collect(Collectors.toList()));
141     }
142 }
