31  import java.io.Serializable;
32  import java.util.stream.Collectors;
33  
34  import javax.persistence.EntityManager;
35  import javax.persistence.PersistenceContext;
36  
37  @RestController
38  @Validated
39  @RequestMapping(
40          path = "/app/{appId}/layer/{appLayerId}/describe",
41          produces = MediaType.APPLICATION_JSON_VALUE)
42  public class LayerDescriptionController {
43      private final ApplicationRepository applicationRepository;
44      private final ApplicationLayerRepository applicationLayerRepository;
45      private final AuthorizationService authorizationService;
46  
47      @PersistenceContext private EntityManager entityManager;
48  
49      @Autowired
50      public LayerDescriptionController(
51              ApplicationRepository applicationRepository,
52              ApplicationLayerRepository applicationLayerRepository,
53              AuthorizationService authorizationService) {
54          this.applicationRepository = applicationRepository;
55          this.applicationLayerRepository = applicationLayerRepository;
56          this.authorizationService = authorizationService;
57      }
58  
59      @GetMapping
60      public ResponseEntity<Serializable> getAppLayerDescription(
61              @Parameter(name = "appId", description = "application id", required = true)
62                      @PathVariable("appId")
63                      Long appId,
64              @Parameter(name = "appLayerId", description = "application layer id", required = true)
65                      @PathVariable("appLayerId")
66                      Long appLayerId) {
67          final Application application = applicationRepository.findById(appId).orElse(null);
68          final ApplicationLayer appLayer =
69                  applicationLayerRepository.findById(appLayerId).orElse(null);
70          if (application == null || appLayer == null) {
71              return ResponseEntity.status(HttpStatus.NOT_FOUND)
72                      .body("Requested an application or appLayer that does not exist");
73          }
74  
75          if (!authorizationService.mayUserRead(appLayer, application)) {
76              return ResponseEntity.status(HttpStatus.UNAUTHORIZED).body(new RedirectResponse());
77          }
78  
79          final Layer layer = appLayer.getService().getLayer(appLayer.getLayerName(), entityManager);
80          if (layer == null) {
81              return ResponseEntity.status(HttpStatus.NOT_FOUND)
82                      .body(
83                              String.format(
84                                      "Can't find layer in service #%d with name \"%s\"",
85                                      appLayer.getService().getId(), appLayer.getLayerName()));
86          }
87  
88          final SimpleFeatureType sft = layer.getFeatureType();
89          if (sft == null) {
90              return ResponseEntity.status(HttpStatus.NOT_FOUND)
91                      .body("Layer does not have feature type");
92          }
93  
94          LayerDetails r = new LayerDetails();
95          r.setId(appLayerId);
96          r.setServiceId(appLayer.getService().getId());
97          r.setFeatureTypeName(sft.getTypeName());
98          r.setGeometryAttribute(sft.getGeometryAttribute());
99          r.attributes(
100                 sft.getAttributes().stream()
101                         .map(
102                                 ad -> {
103                                     Attribute a = new Attribute();
104                                     a.setId(ad.getId());
105                                     a.setName(ad.getName());
106                                     a.setAlias(ad.getAlias());
107                                     a.setType(Attribute.TypeEnum.fromValue(ad.getType()));
108                                     return a;
109                                 })
110                         .collect(Collectors.toList()));
111         return ResponseEntity.ok(r);
112     }
113 }
