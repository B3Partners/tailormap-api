6   package nl.b3p.tailormap.api.security;
7   
8   import org.springframework.context.annotation.Bean;
9   import org.springframework.context.annotation.Configuration;
10  import org.springframework.security.config.annotation.web.builders.HttpSecurity;
11  import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
12  import org.springframework.security.web.SecurityFilterChain;
13  import org.springframework.security.web.csrf.CookieCsrfTokenRepository;
14  
15  @Configuration
16  @EnableWebSecurity
17  public class SecurityConfig {
18      @Bean
19      public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
20          // Note: CSRF protection only required when using cookies for authentication
21          // This requires an X-XSRF-TOKEN header read from the XSRF-TOKEN cookie by JavaScript so set
22          // HttpOnly to false.
23          // Angular has automatic XSRF protection support:
24          // https://angular.io/guide/http#security-xsrf-protection
25          CookieCsrfTokenRepository csrfTokenRepository =
26                  CookieCsrfTokenRepository.withHttpOnlyFalse();
27          csrfTokenRepository.setCookiePath("/");
28          http.csrf()
29                  .csrfTokenRepository(csrfTokenRepository)
30                  .and()
31                  .authorizeHttpRequests()
32                  .requestMatchers("/admin/**")
33                  .hasRole("ADMIN")
34                  .anyRequest()
35                  .permitAll()
36                  .and()
37                  .formLogin();
38          return http.build();
39      }
40  }
