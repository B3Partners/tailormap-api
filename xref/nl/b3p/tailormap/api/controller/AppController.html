41  @RestController
42  @CrossOrigin
43  @RequestMapping(path = "/app")
44  public class AppController {
45      private final Log logger = LogFactory.getLog(getClass());
46  
47      private final ApplicationRepository applicationRepository;
48      private final MetadataRepository metadataRepository;
49  
50      @Value("${tailormap-api.apiVersion}")
51      private String apiVersion;
52  
53      public AppController(
54              ApplicationRepository applicationRepository, MetadataRepository metadataRepository) {
55          this.applicationRepository = applicationRepository;
56          this.metadataRepository = metadataRepository;
57      }

59      /**
60       * Handle any {@code TailormapConfigurationException} that this controller might throw while
61       * getting the application.
62       *
63       * @param exception the exception
64       * @return an error response
65       */
66      @ExceptionHandler(TailormapConfigurationException.class)
67      @ResponseStatus(value = HttpStatus.INTERNAL_SERVER_ERROR)
68      @ResponseBody
69      public ErrorResponse handleTailormapConfigurationException(
70              TailormapConfigurationException exception) {
71          logger.fatal(exception.getMessage());
72          return new ErrorResponse()
73                  .message("Internal server error. " + exception.getMessage())
74                  .code(500);
75      }
76  
77      /**
78       * Lookup an {@linkplain Application} with given parameters. Use this endpoint to get the id of
79       * the requested or default application. Either call this with `name` and optional `version` or
80       * `appId` alone. Will return general setup information such as name, appId, language, but not
81       * map specific information, may return a redirect response for login.
82       *
83       * @param appId the unique identifier of an app
84       * @param name the name of an app
85       * @param version the version of an app
86       * @return the basic information needed to create an app in the frontend
87       * @since 0.1
88       * @throws TailormapConfigurationException when the tailormap configuration is broken
89       */
90      @GetMapping(produces = MediaType.APPLICATION_JSON_VALUE)
91      @Timed(value = "get_app", description = "time spent to find an app")
92      public ResponseEntity<Serializable> get(
93              @RequestParam(required = false) Long appId,
94              @RequestParam(required = false) String name,
95              @RequestParam(required = false) String version)
96              throws TailormapConfigurationException {
97  
98          logger.trace(
99                  "requested application using: appId: "
100                         + appId
101                         + ", name: "
102                         + name
103                         + ", version: "
104                         + version);
105 
106         Application application;
107         if (null != appId) {
108             application = applicationRepository.findById(appId).orElse(null);
109         } else {
110             application = findApplication(name, version);
111         }
112 
113         if (null == application) {
114             application = getDefaultViewer();
115         }
116 
117         if (null == application) {
118             // no default application or something else is very wrong
119             throw new TailormapConfigurationException(
120                     "Error getting the requested or default application.");
121         } else if (application.isAuthenticatedRequired() && !AuthUtil.isAuthenticatedUser()) {
122             // login required, send RedirectResponse
123             return ResponseEntity.status(HttpStatus.UNAUTHORIZED).body(new RedirectResponse());
124         } else {
125             logger.trace(
126                     "found application - id:"
127                             + application.getId()
128                             + ", name: "
129                             + application.getName()
130                             + ", version: "
131                             + application.getVersion()
132                             + ", lang: "
133                             + application.getLang()
134                             + ", title: "
135                             + application.getTitle());
136 
137             AppResponse appResponse =
138                     new AppResponse()
139                             .apiVersion(this.apiVersion)
140                             .id(application.getId())
141                             .name(application.getName())
142                             // any of these 2 below + language could be null
143                             .version(application.getVersion())
144                             .title(application.getTitle());
145 
146             // null check language because it's an enumerated value
147             if (null != application.getLang())
148                 appResponse.lang(AppResponse.LangEnum.fromValue(application.getLang()));
149 
150             return ResponseEntity.ok(appResponse);
151         }
152     }
153 
154     /**
155      * recursive method to find application by name and version.
156      *
157      * @param name user given name
158      * @param version user given version (can be {@code null})
159      * @return found Application (or {@code null} if nothing found)
160      */
161     private Application findApplication(String name, String version) {
162         Application application = null;
163         if (name != null) {
164             if (null != version) {
165                 application = applicationRepository.findByNameAndVersion(name, version);
166             } else {
167                 application =
168                         applicationRepository.findByName(name).stream()
169                                 .sorted()
170                                 .findFirst()
171                                 .orElse(null);
172             }
173 
174             if (null == application) {
175                 logger.warn("No application found with name " + name + " and version " + version);
176                 String decodedName = URLDecoder.decode(name, StandardCharsets.UTF_8);
177                 if (!decodedName.equals(name)) {
178                     return findApplication(decodedName, version);
179                 }
180             }
181         }
182         return application;
183     }
184 
185     /**
186      * find the default viewer (name and version) in this instance.
187      *
188      * @return either the default Application or {@code null} if not configured
189      */
190     private Application getDefaultViewer() {
191         try {
192             Metadata md = metadataRepository.findByConfigKey(Metadata.DEFAULT_APPLICATION);
193             String appId = md.getConfigValue();
194             Long id = Long.parseLong(appId);
195             return applicationRepository.getReferenceById(id);
196         } catch (NullPointerException | EntityNotFoundException e) {
197             logger.warn("No default application configured. " + e.getMessage());
198             return null;
199         }
200     }
