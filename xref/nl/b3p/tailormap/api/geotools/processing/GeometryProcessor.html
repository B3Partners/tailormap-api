6   package nl.b3p.tailormap.api.geotools.processing;
7   
8   import org.apache.commons.logging.Log;
9   import org.apache.commons.logging.LogFactory;
10  import org.locationtech.jts.geom.Geometry;
11  import org.locationtech.jts.geom.PrecisionModel;
12  import org.locationtech.jts.precision.GeometryPrecisionReducer;
13  import org.locationtech.jts.simplify.TopologyPreservingSimplifier;
14  
15  import java.nio.charset.StandardCharsets;
16  
17  import javax.validation.constraints.NotNull;
18  
19  /**
20   * Utility functions on feature geometries.
21   *
22   * @author mprins
23   * @since 0.1
24   */
25  public final class GeometryProcessor {
26      private static final Log LOG = LogFactory.getLog(GeometryProcessor.class);
27  
28      private GeometryProcessor() {}
29  
30      /**
31       * process the geometry into a (optionally simplified) string representation.
32       *
33       * @param geometry An object representing a geometry
34       * @param simplfyGeometry set to {@code true} to simplify
35       * @return the string representation of the argument - normally WKT, optionally simplified
36       */
37      @NotNull
38      public static String processGeometry(
39              @NotNull final Object geometry, @NotNull final Boolean simplfyGeometry) {
40          if (simplfyGeometry && Geometry.class.isAssignableFrom(geometry.getClass())) {
41              return simplify((Geometry) geometry);
42          }
43          // return WKT
44          return geometry.toString();
45      }
46  
47      /**
48       * Simplifies given geometry to reduce (transfer) size, start off with 1 and each iteration
49       * multiply with 10, max 4 steps, so [1, 10, 100, 1000], if the geomeomtry is still too large
50       * bail out and use bbox. TODO this works for CRS in meters, may not work for degrees
51       *
52       * @param geom gemetry to simplify
53       * @return simplified geometry as WKT string
54       */
55      @NotNull
56      private static String simplify(@NotNull Geometry geom) {
57          final int megabytes = (2097152 /* 2MB is the default tomcat max post size */ - 100 * 1024);
58  
59          LOG.debug(geom.getPrecisionModel().getScale());
60          PrecisionModel pm = new PrecisionModel(geom.getPrecisionModel());
61          GeometryPrecisionReducer gpr = new GeometryPrecisionReducer(pm);
62          geom = gpr.reduce(geom);
63          Geometry bbox = geom.getEnvelope();
64  
65          double distanceTolerance = 1.0;
66          String geomTxt = geom.toText();
67  
68          while ((geomTxt.getBytes(StandardCharsets.UTF_8).length > megabytes
69                          || geom.getCoordinates().length > 600)
70                  && distanceTolerance < 9999) {
71              LOG.debug("Simplify selected feature geometry with distance of: " + distanceTolerance);
72              geom = TopologyPreservingSimplifier.simplify(geom, distanceTolerance);
73              geom = gpr.reduce(geom);
74              geomTxt = geom.toText();
75              distanceTolerance = 10 * distanceTolerance;
76          }
77  
78          if (distanceTolerance > 9999) {
79              LOG.debug("Maximum number of simplify cycles reached, returning bounding box instead.");
80              return bbox.toText();
81          } else {
82              return geomTxt;
83          }
84      }
85  }
