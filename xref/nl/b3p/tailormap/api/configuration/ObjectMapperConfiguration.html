6   package nl.b3p.tailormap.api.configuration;
7   
8   import com.fasterxml.jackson.databind.InjectableValues;
9   import com.fasterxml.jackson.databind.ObjectMapper;
10  import org.springframework.beans.factory.annotation.Value;
11  import org.springframework.context.annotation.Bean;
12  import org.springframework.context.annotation.Configuration;
13  import org.springframework.context.annotation.Primary;
14  
15  @Configuration
16  public class ObjectMapperConfiguration {
17    private static final org.slf4j.Logger logger =
18        org.slf4j.LoggerFactory.getLogger(java.lang.invoke.MethodHandles.lookup().lookupClass());
19  
20    @Bean
21    @Primary
22    public ObjectMapper mapper(
23        @Value("${tailormap-api.strong-password.validation:true}") boolean enabled,
24        @Value("${tailormap-api.strong-password.min-length:8}") int minLength,
25        @Value("${tailormap-api.strong-password.min-strength:4}") int minStrength) {
26  
27      logger.info(
28          String.format(
29              "Configuring strong-password policy: enabled=%s, minLength=%s, minStrength=%s",
30              enabled, minLength, minStrength));
31      InjectableValues.Std std = new InjectableValues.Std();
32      std.addValue("tailormap-api.strong-password.validation", enabled);
33      std.addValue("tailormap-api.strong-password.min-length", minLength);
34      std.addValue("tailormap-api.strong-password.min-strength", minStrength);
35  
36      ObjectMapper mapper = new ObjectMapper();
37      mapper.setInjectableValues(std);
38      return mapper;
39    }
40  }
