9   import static org.junit.jupiter.api.Assertions.assertNotNull;

12  import org.junit.jupiter.api.BeforeAll;
13  import org.junit.jupiter.api.Test;
14  import org.springframework.beans.factory.annotation.Autowired;
15  import org.springframework.boot.test.context.SpringBootTest;

17  import java.io.IOException;
18  
19  // this test could just as well be a (unit)"Test" instead of an (i-test)"IntegrationTest", but we
20  // wanted
21  // to have an integration test to check if that works from Maven.
22  @SpringBootTest
23  class VersionControllerIntegrationTest {
24      private static String projectVersion;
25      @Autowired VersionController versionController;
26  
27      @BeforeAll
28      static void getVersionFromPom() {
29          // set through maven surefire/failsafe plugin from pom
30          projectVersion = System.getProperty("project.version");
31          assumeFalse(
32                  null == projectVersion,
33                  "Project version unknown, should be set in system environment");
34      }
35  
36      @Test
37      void testGetVersion() throws IOException {
38          assertNotNull(versionController, "versionController can not be `null` if SpringBoot works");
39          assertNotNull(versionController.getVersion(), "Project version not found");
40          assertEquals(projectVersion, versionController.getVersion(), "Project version incorrect");
41      }
42  }
