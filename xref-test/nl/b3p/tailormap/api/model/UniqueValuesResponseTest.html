13  import java.util.Set;
14  
15  class UniqueValuesResponseTest {
16  
17      @Test
18      void testUniqueValuesResponse() {
19          UniqueValuesResponse uniqueValuesResponse = new UniqueValuesResponse().filterApplied(false);
20          uniqueValuesResponse.addValuesItem("value 1");
21          uniqueValuesResponse.addValuesItem("value 2");
22          uniqueValuesResponse.addValuesItem("value 2");
23          assertEquals(
24                  Set.of("value 1", "value 2"),
25                  uniqueValuesResponse.getValues(),
26                  "values don't match");
27          assertFalse(uniqueValuesResponse.getFilterApplied(), "filterApplied should be true");
28      }
