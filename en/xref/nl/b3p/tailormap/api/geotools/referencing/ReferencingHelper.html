6   package nl.b3p.tailormap.api.geotools.referencing;
7   
8   import nl.b3p.tailormap.api.model.Bounds;
9   
10  import org.apache.commons.logging.Log;
11  import org.apache.commons.logging.LogFactory;
12  import org.geotools.referencing.CRS;
13  import org.opengis.geometry.Envelope;
14  import org.opengis.referencing.FactoryException;
15  import org.opengis.referencing.crs.CoordinateReferenceSystem;
16  
17  public final class ReferencingHelper {
18      private static final Log logger = LogFactory.getLog(ReferencingHelper.class);
19  
20      /**
21       * Try to get the envelope from the CRS.
22       *
23       * @param authorityCode epsg code
24       * @return the bounds, possibly with unknown extents
25       * @see CRS#getEnvelope(CoordinateReferenceSystem)
26       */
27      public static Bounds crsBoundsExtractor(String authorityCode) {
28          try {
29              final CoordinateReferenceSystem crs = CRS.decode(authorityCode);
30              Envelope envelope = CRS.getEnvelope(crs);
31              // envelope can be null
32              if (null != envelope) {
33                  return new Bounds()
34                          .crs(authorityCode)
35                          // ordinate choice may not always be correct...eg. with flipped axis
36                          .maxx(envelope.getUpperCorner().getOrdinate(0))
37                          .maxy(envelope.getUpperCorner().getOrdinate(1))
38                          .minx(envelope.getLowerCorner().getOrdinate(0))
39                          .miny(envelope.getLowerCorner().getOrdinate(1));
40              }
41          } catch (FactoryException e) {
42              logger.warn("Failed to decode CRS. " + e.getMessage(), e);
43          }
44          return new Bounds().crs(authorityCode);
45      }
46  }
