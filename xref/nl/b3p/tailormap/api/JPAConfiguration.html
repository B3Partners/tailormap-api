6   package nl.b3p.tailormap.api;
7   
8   import org.apache.commons.logging.Log;
9   import org.apache.commons.logging.LogFactory;
10  import org.springframework.beans.factory.annotation.Autowired;
11  import org.springframework.boot.autoconfigure.domain.EntityScan;
12  import org.springframework.context.annotation.Bean;
13  import org.springframework.context.annotation.Configuration;
14  import org.springframework.core.env.Environment;
15  import org.springframework.dao.annotation.PersistenceExceptionTranslationPostProcessor;
16  import org.springframework.data.jpa.repository.config.EnableJpaRepositories;
17  import org.springframework.jdbc.datasource.DriverManagerDataSource;
18  import org.springframework.orm.jpa.JpaTransactionManager;
19  import org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean;
20  import org.springframework.orm.jpa.vendor.HibernateJpaVendorAdapter;
21  import org.springframework.transaction.annotation.EnableTransactionManagement;
22  
23  import java.util.Objects;
24  import java.util.Properties;
25  
26  import javax.persistence.EntityManagerFactory;
27  import javax.sql.DataSource;
28  
29  /**
30   * JPA configuration beans.
31   *
32   * @since 0.1
33   */
34  @Configuration
35  @EnableJpaRepositories(basePackages = {"nl.b3p.tailormap.api.repository"})
36  @EntityScan(
37          basePackages = {
38              "nl.tailormap.viewer.config",
39              "nl.tailormap.viewer.config.app",
40              "nl.tailormap.viewer.config.metadata",
41              "nl.tailormap.viewer.config.security",
42              "nl.tailormap.viewer.config.services"
43          })
44  @EnableTransactionManagement
45  public class JPAConfiguration {
46      @Autowired private Environment env;
47  
48      private final Log logger = LogFactory.getLog(getClass());
49  
50      @Bean
51      public LocalContainerEntityManagerFactoryBean entityManagerFactory() {
52          final LocalContainerEntityManagerFactoryBean em =
53                  new LocalContainerEntityManagerFactoryBean();
54  
55          em.setJpaVendorAdapter(new HibernateJpaVendorAdapter());
56          em.setJpaProperties(additionalProperties());
57          em.setPersistenceUnitName("viewer-config-postgresql");
58          em.setDataSource(dataSource());
59  
60          return em;
61      }
62  
63      /**
64       * This is to override the config of the provided persistence module.
65       *
66       * @return configured datasource
67       */
68      @Bean
69      public DataSource dataSource() {
70          final DriverManagerDataSource dataSource = new DriverManagerDataSource();
71          dataSource.setUrl(env.getProperty("spring.datasource.url"));
72          dataSource.setDriverClassName(
73                  Objects.requireNonNull(env.getProperty("spring.datasource.driver-class-name")));
74          dataSource.setUsername(env.getProperty("spring.datasource.username"));
75          dataSource.setPassword(env.getProperty("spring.datasource.password"));
76  
77          logger.debug(
78                  "using database: "
79                          + env.getProperty("spring.datasource.url")
80                          + " with driver: "
81                          + env.getProperty("spring.datasource.driver-class-name"));
82          return dataSource;
83      }
84  
85      @Bean
86      JpaTransactionManager transactionManager(final EntityManagerFactory entityManagerFactory) {
87          final JpaTransactionManager transactionManager = new JpaTransactionManager();
88          transactionManager.setEntityManagerFactory(entityManagerFactory);
89  
90          return transactionManager;
91      }
92  
93      @Bean
94      public PersistenceExceptionTranslationPostProcessor exceptionTranslation() {
95          return new PersistenceExceptionTranslationPostProcessor();
96      }
97  
98      private final Properties additionalProperties() {
99          final Properties hibernateProperties = new Properties();
100         //    hibernateProperties.setProperty(
101         //        "hibernate.hbm2ddl.auto", env.getProperty("spring.jpa.hibernate.ddl-auto", ""));
102         hibernateProperties.setProperty(
103                 "hibernate.show_sql", env.getProperty("spring.jpa.show-sql", "false"));
104         hibernateProperties.setProperty(
105                 "hibernate.generate_statistics",
106                 env.getProperty("spring.jpa.properties.hibernate.enable_metrics", "true"));
107 
108         return hibernateProperties;
109     }
110 }
