379      * @throws Exception if any

381     @Test
382     void should_return_non_empty_featurecollections_for_valid_page_from_wfs() throws Exception {
383         // bestuurlijke gebieden WFS; provincies
384         // page 1
385         MvcResult result =
386                 mockMvc.perform(get(provinciesWFS).param("page", "1"))
387                         .andExpect(status().isOk())
388                         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
389                         .andExpect(
390                                 jsonPath("$.total")
391                                         .value(exactWfsCounts ? provinciesWFSTotalCount : -1))
392                         .andExpect(jsonPath("$.page").value(1))
393                         .andExpect(jsonPath("$.pageSize").value(pageSize))
394                         .andExpect(jsonPath("$.features").isArray())
395                         .andExpect(jsonPath("$.features").isNotEmpty())
396                         .andExpect(jsonPath("$.features[0]").isMap())
397                         .andExpect(jsonPath("$.features[0]").isNotEmpty())
398                         .andExpect(jsonPath("$.features[0].__fid").isNotEmpty())
399                         .andExpect(jsonPath("$.features[0].geometry").isEmpty())
400                         .andExpect(jsonPath("$.features[0].attributes.naam").value("Drenthe"))
401                         .andExpect(
402                                 jsonPath("$.features[0].attributes.ligtInLandNaam")
403                                         .value("Nederland"))
404                         .andExpect(jsonPath("$.columnMetadata").isArray())
405                         .andExpect(jsonPath("$.columnMetadata").isNotEmpty())
406                         .andReturn();
407 
408         String body = result.getResponse().getContentAsString();
409         logger.trace(body);
410         assertNotNull(body, "response body should not be null");
411         List<Service> page1Features = JsonPath.read(body, "$.features");
412         assertEquals(
413                 pageSize,
414                 page1Features.size(),
415                 () ->
416                         "there should be "
417                                 + pageSize
418                                 + " provinces in the list, but was "
419                                 + page1Features.size());
420 
421         // page 2
422         result =
423                 // bestturlijke gebieden WFS; provincies
424                 mockMvc.perform(get(provinciesWFS).param("page", "2"))
425                         .andExpect(status().isOk())
426                         .andExpect(content().contentType(MediaType.APPLICATION_JSON))
427                         .andExpect(
428                                 jsonPath("$.total")
429                                         .value(exactWfsCounts ? provinciesWFSTotalCount : -1))
430                         .andExpect(jsonPath("$.page").value(2))
431                         .andExpect(jsonPath("$.pageSize").value(pageSize))
432                         .andExpect(jsonPath("$.features").isArray())
433                         .andExpect(jsonPath("$.features").isNotEmpty())
434                         .andExpect(jsonPath("$.columnMetadata").isArray())
435                         .andExpect(jsonPath("$.columnMetadata").isNotEmpty())
436                         .andReturn();
437 
438         body = result.getResponse().getContentAsString();
439         logger.trace(body);
440         assertNotNull(body, "response body should not be null");
441 
442         List<Service> page2Features = JsonPath.read(body, "$.features");
443         assertEquals(
444                 2,
445                 page2Features.size(),
446                 () -> "there should be 2 provinces in the list, but was " + page2Features.size());
447 
448         // check for duplicates
449         int retrievedFeatsCount = page1Features.size() + page2Features.size();
450         page2Features.addAll(page1Features);
451         assertEquals(
452                 retrievedFeatsCount,
453                 (int) page2Features.stream().distinct().count(),
454                 "there should be no duplicates in 2 sequential pages");
455     }
