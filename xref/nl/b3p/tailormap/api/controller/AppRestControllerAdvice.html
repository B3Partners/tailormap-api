6   package nl.b3p.tailormap.api.controller;
7   
8   import nl.b3p.tailormap.api.annotation.AppRestController;
9   import nl.b3p.tailormap.api.model.ErrorResponse;
10  import nl.b3p.tailormap.api.model.RedirectResponse;
11  import nl.b3p.tailormap.api.repository.ApplicationLayerRepository;
12  import nl.b3p.tailormap.api.repository.ApplicationRepository;
13  import nl.b3p.tailormap.api.security.AuthorizationService;
14  import nl.tailormap.viewer.config.app.Application;
15  import nl.tailormap.viewer.config.app.ApplicationLayer;
16  
17  import org.springframework.http.HttpStatus;
18  import org.springframework.http.MediaType;
19  import org.springframework.http.ResponseEntity;
20  import org.springframework.web.bind.WebDataBinder;
21  import org.springframework.web.bind.annotation.ExceptionHandler;
22  import org.springframework.web.bind.annotation.InitBinder;
23  import org.springframework.web.bind.annotation.ModelAttribute;
24  import org.springframework.web.bind.annotation.PathVariable;
25  import org.springframework.web.bind.annotation.RestControllerAdvice;
26  import org.springframework.web.server.ResponseStatusException;
27  
28  @RestControllerAdvice(annotations = AppRestController.class)
29  public class AppRestControllerAdvice {
30  
31      private final ApplicationRepository applicationRepository;
32      private final ApplicationLayerRepository applicationLayerRepository;
33      private final AuthorizationService authorizationService;
34  
35      public AppRestControllerAdvice(
36              ApplicationRepository applicationRepository,
37              ApplicationLayerRepository applicationLayerRepository,
38              AuthorizationService authorizationService) {
39          this.applicationRepository = applicationRepository;
40          this.applicationLayerRepository = applicationLayerRepository;
41          this.authorizationService = authorizationService;
42      }
43  
44      @InitBinder
45      protected void initBinder(WebDataBinder binder) {
46          binder.setAllowedFields("appId", "appLayerId");
47      }
48  
49      @ExceptionHandler(ResponseStatusException.class)
50      protected ResponseEntity<?> handleResponseStatusException(ResponseStatusException ex) {
51          if (HttpStatus.UNAUTHORIZED.equals(ex.getStatus())) {
52              return ResponseEntity.status(HttpStatus.UNAUTHORIZED)
53                      .contentType(MediaType.APPLICATION_JSON)
54                      .body(new RedirectResponse());
55          }
56          return ResponseEntity.status(ex.getRawStatusCode())
57                  .contentType(MediaType.APPLICATION_JSON)
58                  .body(new ErrorResponse().message(ex.getReason()).code(ex.getRawStatusCode()));
59      }
60  
61      @ModelAttribute
62      public Application populateApplication(@PathVariable(required = false) Long appId) {
63          if (appId == null) {
64              // No binding for AppController which will return the default application instead of 404
65              return null;
66          }
67          final Application application = applicationRepository.findById(appId).orElse(null);
68          if (application == null) {
69              throw new ResponseStatusException(
70                      HttpStatus.NOT_FOUND, "Application with id " + appId + " not found");
71          }
72          if (!this.authorizationService.mayUserRead(application)) {
73              throw new ResponseStatusException(HttpStatus.UNAUTHORIZED);
74          }
75          return application;
76      }
77  
78      @ModelAttribute
79      public ApplicationLayer populateApplicationLayer(
80              @ModelAttribute Application application,
81              @PathVariable(required = false) Long appLayerId) {
82          if (appLayerId == null) {
83              // No binding
84              return null;
85          }
86  
87          final ApplicationLayer applicationLayer =
88                  applicationLayerRepository.findById(appLayerId).orElse(null);
89          if (applicationLayer == null) {
90              throw new ResponseStatusException(
91                      HttpStatus.NOT_FOUND, "Application layer with id " + appLayerId + " not found");
92          }
93  
94          // Also verifies that the application layer actually belongs to the application
95          if (!this.authorizationService.mayUserRead(applicationLayer, application)) {
96              throw new ResponseStatusException(HttpStatus.FORBIDDEN, "Access denied");
97          }
98          return applicationLayer;
99      }
100 }
