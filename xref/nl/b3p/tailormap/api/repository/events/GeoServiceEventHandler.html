6   package nl.b3p.tailormap.api.repository.events;
7   
8   import io.hypersistence.tsid.TSID;
9   import nl.b3p.tailormap.api.persistence.GeoService;
10  import nl.b3p.tailormap.api.persistence.helper.GeoServiceHelper;
11  import org.apache.commons.lang3.StringUtils;
12  import org.springframework.data.rest.core.annotation.HandleBeforeCreate;
13  import org.springframework.data.rest.core.annotation.RepositoryEventHandler;
14  import org.springframework.stereotype.Component;
15  
16  @RepositoryEventHandler
17  @Component
18  public class GeoServiceEventHandler {
19    private final GeoServiceHelper geoServiceHelper;
20  
21    public GeoServiceEventHandler(GeoServiceHelper geoServiceHelper) {
22      this.geoServiceHelper = geoServiceHelper;
23    }
24  
25    @HandleBeforeCreate
26    public void assignId(GeoService geoService) {
27      if (StringUtils.isBlank(geoService.getId())) {
28        // We kind of misuse TSIDs here, because we store it as a string. This is because the id
29        // string can also be manually assigned. There won't be huge numbers of GeoServices, so it's
30        // more of a convenient way to generate an ID that isn't a huge UUID string.
31        geoService.setId(TSID.fast().toString());
32      }
33    }
34  
35    @HandleBeforeCreate
36    public void loadCapabilities(GeoService geoService) throws Exception {
37      geoServiceHelper.loadServiceCapabilities(geoService);
38    }
39  }
