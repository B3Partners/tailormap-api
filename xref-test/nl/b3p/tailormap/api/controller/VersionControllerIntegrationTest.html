22  import java.util.Map;

24  /** Testcases for {@link VersionController}. */
25  @SpringBootTest(classes = {HSQLDBTestProfileJPAConfiguration.class, VersionController.class})
26  @ActiveProfiles("test")
27  class VersionControllerIntegrationTest {
28      @Autowired private VersionController versionController;
29      @Autowired private MetadataRepository metadataRepository;
30  
31      @Test
32      void testGetVersion() {
33          assertNotNull(
34                  versionController, "versionController can not be `null` if Spring Boot works");
35          assertNotNull(versionController.getVersion(), "Version info not found");
36  
37          String projectVersion = System.getProperty("project.version");
38          assumeFalse(
39                  null == projectVersion,
40                  "Project version unknown, should be set as system property");
41          String databaseVersion = System.getenv("DATABASE_VERSION");
42          assumeFalse(
43                  null == databaseVersion,
44                  "Database version unknown, should be set in system environment");
45  
46          String apiVersion = System.getenv("API_VERSION");
47          assumeFalse(null == apiVersion, "API version unknown, should be set in system environment");
48  
49          Map<String, String> expected =
50                  Map.of(
51                          "version",
52                          projectVersion,
53                          "databaseversion",
54                          databaseVersion,
55                          "apiVersion",
56                          apiVersion);
57  
58          assertEquals(expected, versionController.getVersion(), "Unexpected version response.");
59      }
60  
61      @Test
62      /* this test changes database content */
63      @Order(Integer.MAX_VALUE)
64      void unknown_database_version() throws Exception {
65          metadataRepository.deleteMetadataByConfigKey(Metadata.DATABASE_VERSION_KEY);
66  
67          String projectVersion = System.getProperty("project.version");
68          assumeFalse(
69                  null == projectVersion,
70                  "Project version unknown, should be set as system property");
71  
72          String apiVersion = System.getenv("API_VERSION");
73          assumeFalse(null == apiVersion, "API version unknown, should be set in system environment");
74  
75          assertNotNull(versionController.getVersion(), "Version info not found");
76          Map<String, String> expected =
77                  Map.of(
78                          "version",
79                          projectVersion,
80                          "databaseversion",
81                          "unknown",
82                          "apiVersion",
83                          apiVersion);
84  
85          assertEquals(expected, versionController.getVersion(), "Unexpected version response.");
86      }
87  }
