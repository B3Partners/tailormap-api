7   package nl.b3p.tailormap.api.configuration;
8   
9   import java.util.List;
10  import javax.annotation.PostConstruct;
11  import nl.b3p.tailormap.api.persistence.Catalog;
12  import nl.b3p.tailormap.api.persistence.json.CatalogNode;
13  import nl.b3p.tailormap.api.repository.CatalogRepository;
14  import nl.b3p.tailormap.api.security.InternalAdminAuthentication;
15  import org.springframework.context.annotation.Configuration;
16  import org.springframework.stereotype.Service;
17  
18  @Configuration
19  @Service("tailormap-database-initialization")
20  public class TailormapDatabaseInitialization {
21    CatalogRepository catalogRepository;
22  
23    public TailormapDatabaseInitialization(CatalogRepository catalogRepository) {
24      this.catalogRepository = catalogRepository;
25    }
26  
27    @PostConstruct
28    public void databaseInitialization() {
29      InternalAdminAuthentication.setInSecurityContext();
30      try {
31        if (catalogRepository.findById(Catalog.MAIN).isEmpty()) {
32          Catalog catalog =
33              new Catalog()
34                  .setId(Catalog.MAIN)
35                  .setNodes(List.of(new CatalogNode().root(true).title("root").id("root")));
36          catalogRepository.save(catalog);
37        }
38      } finally {
39        InternalAdminAuthentication.clearSecurityContextAuthentication();
40      }
41    }
42  }
