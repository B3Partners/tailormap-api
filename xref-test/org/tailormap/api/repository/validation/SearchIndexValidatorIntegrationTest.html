6   package org.tailormap.api.repository.validation;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertFalse;
10  import static org.junit.jupiter.api.Assertions.assertTrue;
11  
12  import org.junit.jupiter.api.BeforeEach;
13  import org.junit.jupiter.api.Test;
14  import org.springframework.beans.factory.annotation.Autowired;
15  import org.springframework.validation.BeanPropertyBindingResult;
16  import org.springframework.validation.Errors;
17  import org.tailormap.api.annotation.PostgresIntegrationTest;
18  import org.tailormap.api.persistence.SearchIndex;
19  import org.tailormap.api.repository.FeatureTypeRepository;
20  
21  @PostgresIntegrationTest
22  class SearchIndexValidatorIntegrationTest {
23    private SearchIndexValidator searchIndexValidator;
24  
25    @Autowired private FeatureTypeRepository featureTypeRepository;
26  
27    @BeforeEach
28    void setUp() {
29      searchIndexValidator = new SearchIndexValidator(featureTypeRepository);
30    }
31  
32    @Test
33    void supports() {
34      assertTrue(searchIndexValidator.supports(SearchIndex.class));
35      assertFalse(searchIndexValidator.supports(Object.class));
36    }
37  
38    @Test
39    void testWFSFeatureType() {
40      SearchIndex wfsIndex = new SearchIndex().setFeatureTypeId(15L);
41      Errors errors = new BeanPropertyBindingResult(wfsIndex, "");
42      searchIndexValidator.validate(wfsIndex, errors);
43  
44      assertEquals(1, errors.getErrorCount(), "Expected 1 error");
45      assertEquals(
46          "This feature type is not available for indexing.",
47          errors.getAllErrors().get(0).getDefaultMessage(),
48          "Did not get expected error message");
49    }
50  
51    @Test
52    void testJDBCFeatureType() {
53      SearchIndex wfsIndex = new SearchIndex().setFeatureTypeId(26L);
54      Errors errors = new BeanPropertyBindingResult(wfsIndex, "");
55      searchIndexValidator.validate(wfsIndex, errors);
56  
57      assertEquals(0, errors.getErrorCount(), "Expected no errors");
58    }
59  }
