52  /**
53   * Proxy controller for OGC WMS and WMTS services. Does not attempt to hide the original service
54   * URL. Mostly useful for access to HTTP Basic secured services without sending the credentials to
55   * the client. The access control is handled by Spring Security and the authorizations configured on
56   * the service.
57   *
58   * <p>Only supports GET requests. Does not support CORS, only meant for tailormap-viewer from the
59   * same origin.
60   *
61   * <p>Implementation note: uses the Java 11 HttpClient. Spring cloud gateway can proxy with many
62   * more features but can not be used in a non-reactive application.
63   */
64  @AppRestController
65  @Validated
66  @RequestMapping(path = "/app/{appId}/layer/{appLayerId}/proxy/{protocol}")
67  public class GeoServiceProxyController {
68  
69      private static final Log logger = LogFactory.getLog(GeoServiceProxyController.class);
70  
71      @RequestMapping(method = {GET, POST})
72      public ResponseEntity<?> proxy(
73              @ModelAttribute Application application,
74              @ModelAttribute ApplicationLayer applicationLayer,
75              @PathVariable("protocol") String protocol,
76              HttpServletRequest request) {
77  
78          Predicate<GeoService> serviceValidator;
79          if ("wmts".equals(protocol)) {
80              serviceValidator =
81                      (GeoService gs) ->
82                              TileService.PROTOCOL.equals(gs.getProtocol())
83                                      && TileService.TILING_PROTOCOL_WMTS.equals(
84                                              ((TileService) gs).getTilingProtocol());
85          } else if ("wms".equals(protocol)) {
86              serviceValidator = (GeoService gs) -> WMSService.PROTOCOL.equals(gs.getProtocol());
87          } else {
88              return ResponseEntity.status(HttpStatus.NOT_FOUND)
89                      .body("Unknown proxy protocol: " + protocol);
90          }
91  
92          final GeoService service = applicationLayer.getService();
93  
94          if (!serviceValidator.test(service)) {
95              return ResponseEntity.status(HttpStatus.BAD_REQUEST)
96                      .body("Invalid proxy protocol for service");
97          }
98  
99          if (!Boolean.parseBoolean(
100                 String.valueOf(service.getDetails().get(GeoService.DETAIL_USE_PROXY)))) {
101             return ResponseEntity.status(HttpStatus.FORBIDDEN)
102                     .body("Proxy not enabled for requested service");
103         }
104 
105         final UriComponentsBuilder originalServiceUrl =
106                 UriComponentsBuilder.fromHttpUrl(service.getUrl());
107         // request.getParameterMap() includes parameters from an
108         // application/x-www-form-urlencoded POST body
109         final MultiValueMap<String, String> requestParams =
110                 request.getParameterMap().entrySet().stream()
111                         .map(
112                                 entry ->
113                                         new AbstractMap.SimpleEntry<>(
114                                                 entry.getKey(),
115                                                 Arrays.stream(entry.getValue())
116                                                         .map(
117                                                                 value ->
118                                                                         UriUtils.encode(
119                                                                                 value,
120                                                                                 StandardCharsets
121                                                                                         .UTF_8))
122                                                         .collect(Collectors.toList())))
123                         .collect(
124                                 Collectors.toMap(
125                                         Map.Entry::getKey,
126                                         Map.Entry::getValue,
127                                         (x, y) -> y,
128                                         LinkedMultiValueMap::new));
129         final MultiValueMap<String, String> params =
130                 buildOgcProxyRequestParams(
131                         originalServiceUrl.build(true).getQueryParams(), requestParams);
132         originalServiceUrl.replaceQueryParams(params);
133 
134         return doProxy(
135                 originalServiceUrl.build(true).toUri(),
136                 application,
137                 applicationLayer,
138                 service,
139                 request);
140     }
141 
142     public static MultiValueMap<String, String> buildOgcProxyRequestParams(
143             MultiValueMap<String, String> originalServiceParams,
144             MultiValueMap<String, String> requestParams) {
145         // Start with all the parameters from the request
146         final MultiValueMap<String, String> params = new LinkedMultiValueMap<>(requestParams);
147 
148         // Add original service URL parameters if they are not required OGC params
149         // (case-insensitive) and not already set by request
150         final List<String> ogcParams = List.of(new String[] {"SERVICE", "REQUEST", "VERSION"});
151         for (Map.Entry<String, List<String>> serviceParam : originalServiceParams.entrySet()) {
152             if (!params.containsKey(serviceParam.getKey())
153                     && !ogcParams.contains(serviceParam.getKey().toUpperCase())) {
154                 params.put(serviceParam.getKey(), serviceParam.getValue());
155             }
156         }
157         return params;
158     }

160     private static ResponseEntity<?> doProxy(
161             URI uri,
162             Application application,
163             ApplicationLayer appLayer,
164             GeoService service,
165             HttpServletRequest request) {
166         final HttpClient.Builder builder =
167                 HttpClient.newBuilder().followRedirects(HttpClient.Redirect.NORMAL);
168 
169         // Some HTTP services may not like HTTP/2 (not configurable in admin at the moment)
170         if (Boolean.parseBoolean(String.valueOf(service.getDetails().get("proxy_http1_1")))) {
171             builder.version(HttpClient.Version.HTTP_1_1);
172         }
173 
174         final HttpClient httpClient = builder.build();
175 
176         HttpRequest.Builder requestBuilder = HttpRequest.newBuilder().uri(uri);
177 
178         try {
179             String ip = request.getRemoteAddr();
180             InetAddress inetAddress = InetAddress.getByName(ip);
181 
182             if (inetAddress instanceof Inet6Address) {
183                 // https://stackoverflow.com/questions/33168783/
184                 Inet6Address inet6Address = (Inet6Address) inetAddress;
185                 int scopeId = inet6Address.getScopeId();
186 
187                 if (scopeId > 0) {
188                     ip = inet6Address.getHostName().replaceAll("%" + scopeId + "$", "");
189                 }
190 
191                 // IPv6 address must be bracketed and quoted
192                 ip = "\"[" + ip + "]\"";
193             }
194             requestBuilder.header("X-Forwarded-For", ip);
195             requestBuilder.header("Forwarded", "for=" + ip);
196         } catch (UnknownHostException ignored) {
197             // Don't care
198         }
199 
200         String[] requestHeaders = {
201             "Accept",
202             "If-Modified-Since",
203             "If-Unmodified-Since",
204             "If-Match",
205             "If-None-Match",
206             "If-Range",
207             "Range",
208             "Referer",
209             "User-Agent",
210         };
211         for (String header : requestHeaders) {
212             String value = request.getHeader(header);
213             if (value != null) {
214                 requestBuilder.header(header, value);
215             }
216         }
217 
218         boolean addCredentials = service.getUsername() != null && service.getPassword() != null;
219         if (addCredentials) {
220             // An app admin should not be able to accidentally open a proxy to a secured geo service
221             // in a public app, so only allow proxying to a secured geo service in apps which
222             // require
223             // authentication.
224 
225             // A geo service which is in a private network which does not require credentials /can/
226             // be 'exposed' in a public app through this proxy.
227 
228             // An app may have authenticated required but "no groups checked" to allow anyone to
229             // access the service (after being logged in for the app). This might still be an
230             // inadvertent misconfiguration but less concerning.
231 
232             // Previous tailormap proxy behaviour was to proxy the request but not add the
233             // authentication. This proxy sends a Forbidden response.
234 
235             if (!application.isAuthenticatedRequired()) {
236                 // XXX for tiled services each request logs a warning, may clutter the log...
237                 logger.warn(
238                         String.format(
239                                 "App %s has app layer %s from proxied secured service URL %s (username %s), but app authentication is not required. Denying proxy, even if user is authenticated.",
240                                 application.getId(),
241                                 appLayer.getId(),
242                                 service.getUrl(),
243                                 service.getUsername()));
244                 return ResponseEntity.status(HttpStatus.FORBIDDEN).body("Forbidden");
245             }
246             String toEncode = service.getUsername() + ":" + service.getPassword();
247             requestBuilder.header(
248                     "Authorization",
249                     "Basic "
250                             + Base64.getEncoder()
251                                     .encodeToString(toEncode.getBytes(StandardCharsets.UTF_8)));
252 
253             // TODO: in some cases we might want to forward the authenticated tailormap user or
254             // groups/roles to the service (perhaps in a secret header name) so the service can
255             // apply different authorizations instead of just one account.
256         }
257 
258         try {
259             HttpResponse<InputStream> response =
260                     httpClient.send(
261                             requestBuilder.build(), HttpResponse.BodyHandlers.ofInputStream());
262             // If the server does not accept our credentials it might 'hide' the layer or even send
263             // a 401 Unauthorized status. We do not send the WWW-Authenticate header back, so the
264             // client will get the error but not an authorization popup.
265             // It would be nice if proxy (auth) errors were logged and available in the admin
266             // interface. Currently, a layer will just stop working if the geo service credentials
267             // are changed without updating them in the geo service registry.
268             InputStreamResource body = new InputStreamResource(response.body());
269             org.springframework.http.HttpHeaders headers = new HttpHeaders();
270             // TODO for WMTS, does caching work?
271             String[] allowedResponseHeaders = {
272                 "Content-Type",
273                 "Content-Length",
274                 "Content-Range",
275                 "Content-Disposition",
276                 "Cache-Control",
277                 "Expires",
278                 "Last-Modified",
279                 "ETag",
280                 "Pragma",
281             };
282             for (String header : allowedResponseHeaders) {
283                 headers.addAll(header, response.headers().allValues(header));
284             }
285             return ResponseEntity.status(response.statusCode()).headers(headers).body(body);
286         } catch (Exception e) {
287             return ResponseEntity.status(HttpStatus.BAD_GATEWAY).body("Bad Gateway");
288         }
289     }
