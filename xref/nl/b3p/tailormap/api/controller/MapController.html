105                                 @Content(

115             })
116     @GetMapping(produces = MediaType.APPLICATION_JSON_VALUE)
117     @ResponseBody
118     @Timed(value = "get_map", description = "time spent to process get the map of an application")
119     public ResponseEntity<Serializable> get(
120             @Parameter(name = "appId", description = "application id", required = true)
121                     @PathVariable("appId")
122                     Long appId) {
123         logger.trace("Requesting 'map' for application id: " + appId);
124 
125         Application application = applicationRepository.findWithGeoservicesById(appId);
126         if (application == null) {
127             throw new EntityNotFoundException();
128         } else if (application.isAuthenticatedRequired() && !AuthUtil.isAuthenticatedUser()) {
129             // login required, send RedirectResponse
130             return ResponseEntity.status(HttpStatus.UNAUTHORIZED).body(new RedirectResponse());
131         } else {
132             MapResponse mapResponse = new MapResponse();
133             getApplicationParams(application, mapResponse);
134             getLayers(application, mapResponse);
135 
136             return ResponseEntity.status(HttpStatus.OK).body(mapResponse);
137         }
138     }

154     private void getApplicationParams(@NotNull Application a, @NotNull MapResponse mapResponse) {
155         final String pCode = a.getProjectionCode();
156 
157         CoordinateReferenceSystem c = new CoordinateReferenceSystem();
158         if (null != pCode) {
159             c.code(ParseUtil.parseEpsgCode(pCode)).definition(ParseUtil.parseProjDefintion(pCode));
160         }
161         Bounds maxExtent = new Bounds();
162         if (null != a.getMaxExtent()) {
163             maxExtent
164                     .minx(a.getMaxExtent().getMinx())
165                     .miny(a.getMaxExtent().getMiny())
166                     .maxx(a.getMaxExtent().getMaxx())
167                     .maxy(a.getMaxExtent().getMaxy())
168                     .crs(a.getMaxExtent().getCrs().getName());
169         } else {
170             maxExtent = ReferencingHelper.crsBoundsExtractor(c.getCode());
171         }
172         Bounds initialExtent = new Bounds();
173         if (null != a.getStartExtent()) {
174             initialExtent
175                     .minx(a.getStartExtent().getMinx())
176                     .miny(a.getStartExtent().getMiny())
177                     .maxx(a.getStartExtent().getMaxx())
178                     .maxy(a.getStartExtent().getMaxy())
179                     .crs(a.getStartExtent().getCrs().getName());
180         } else {
181             initialExtent = maxExtent;
182         }
183 
184         mapResponse.crs(c).maxExtent(maxExtent).initialExtent(initialExtent);
185     }

187     private String getNameForAppLayer(
188             @NotNull ApplicationLayer layer, @NotNull List<Layer> layers) {
189         if (ClobElement.isNotBlank(layer.getDetails().get("titleAlias"))) {
190             return layer.getDetails().get("titleAlias").getValue();
191         } else {
192             Layer serviceLayer = null;
193             for (Layer possibleLayer : layers) {
194                 if (possibleLayer.getService().equals(layer.getService())
195                         && Objects.equals(possibleLayer.getName(), layer.getLayerName())) {
196                     serviceLayer = possibleLayer;
197                     break;
198                 }
199             }
200 
201             if (serviceLayer != null) {
202                 return serviceLayer.getDisplayName();
203             } else {
204                 return layer.getLayerName();
205             }
206         }
207     }

209     private static boolean isAuthorized(Set<String> readers, Authentication auth) {
210         if (readers == null || readers.isEmpty()) {
211             return true;
212         }
213 
214         for (String reader : readers) {
215             if (auth.getAuthorities().stream().anyMatch(x -> x.getAuthority().equals(reader))) {
216                 return true;
217             }
218         }
219 
220         return false;
221     }

223     private void getLayers(@NotNull Application a, @NotNull MapResponse mapResponse) {
224         Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
225 
226         LayerTreeNode rootNode = new LayerTreeNode().id("root").root(true).name("Foreground");
227         mapResponse.addLayerTreeNodesItem(rootNode);
228 
229         LayerTreeNode rootBackgroundNode =
230                 new LayerTreeNode().id("rootbg").root(true).name("Background");
231         mapResponse.addBaseLayerTreeNodesItem(rootBackgroundNode);
232 
233         List<Long> ids = new ArrayList<>();
234         Map<Long, List<Level>> levelChildrenMap = new HashMap<>();
235         for (Level level : levelRepository.findByLevelTree(a.getRoot().getId())) {
236             if (level.getParent() != null) {
237                 levelChildrenMap
238                         .computeIfAbsent(level.getParent().getId(), (Long t) -> new ArrayList<>())
239                         .add(level);
240             }
241             ids.add(level.getId());
242         }
243 
244         // Preload the list of levels, layers, and their readers into the repository's cache.
245         // Note that the result of this is not used directly; this merely ensures that
246         // `level.getLayers()` doesn't need another roundtrip later.
247         levelRepository.findWithAuthorizationDataByIdIn(ids);
248 
249         // List of used ApplicationLayer IDs in this map.
250         Set<Long> applicationLayerIds = new HashSet<>();
251 
252         List<StartLayer> startLayers = a.getStartLayers();
253         for (StartLayer startLayer : startLayers) {
254             if (startLayer.isRemoved()) {
255                 continue;
256             }
257 
258             applicationLayerIds.add(startLayer.getApplicationLayer().getId());
259         }
260 
261         // As above, preload the list of ApplicationLayers, their authorization, and details (used
262         // for display name). We already have handles to ApplicationLayers in the StartLayer
263         // objects.
264         applicationLayerRepository.findWithReadersAndDetailsByIdIn(applicationLayerIds);
265 
266         // Map of (authorized) ApplicationLayer ID to StartLayer objects.
267         Map<Long, StartLayer> layerMap = new HashMap<>(startLayers.size());
268         for (StartLayer startLayer : startLayers) {
269             if (startLayer.isRemoved()) {
270                 continue;
271             }
272 
273             ApplicationLayer appLayer = startLayer.getApplicationLayer();
274             if (!isAuthorized(appLayer.getReaders(), authentication)) {
275                 continue;
276             }
277             layerMap.put(appLayer.getId(), startLayer);
278         }
279 
280         // Fetch the list of possible StartLevels, and filter them down to ones that aren't removed
281         // and are assigned an index.
282         List<StartLevel> startLevels = a.getStartLevels();
283         startLevels.removeIf((StartLevel t) -> t.isRemoved() || t.getSelectedIndex() == null);
284         startLevels.sort(Comparator.comparingLong(StartLevel::getSelectedIndex));
285 
286         // Authorized StartLevel IDs
287         Set<Long> visibleLevels = new HashSet<>();
288         List<StartLayer> visibleStartLayers = new ArrayList<>();
289 
290         // Iterate over startLevels twice. Once to figure out visibility and authentication, once to
291         // actually build the returned structure.
292         Deque<Level> levelQueue = new ArrayDeque<>();
293         for (StartLevel l : startLevels) {
294             levelQueue.add(l.getLevel());
295             while (!levelQueue.isEmpty()) {
296                 Level level = levelQueue.pop();
297                 if (visibleLevels.contains(level.getId())) {
298                     continue;
299                 }
300 
301                 if (!isAuthorized(level.getReaders(), authentication)) {
302                     continue;
303                 }
304 
305                 visibleLevels.add(level.getId());
306                 levelQueue.addAll(levelChildrenMap.getOrDefault(level.getId(), List.of()));
307                 for (ApplicationLayer layer : level.getLayers()) {
308                     StartLayer startLayer = layerMap.get(layer.getId());
309                     if (startLayer == null) {
310                         continue;
311                     }
312 
313                     visibleStartLayers.add(startLayer);
314                 }
315             }
316         }
317 
318         // To check for visibility on the GeoService, we need each Layer and their parents. This
319         // could be done with a native query, however, we also need each layer's readers, which
320         // would require another roundtrip. Take the bandwidth hit and fetch all the used
321         // GeoServices' layers.
322         Set<Long> neededServiceIds = new HashSet<>();
323         for (StartLayer l : visibleStartLayers) {
324             neededServiceIds.add(l.getApplicationLayer().getService().getId());
325         }
326         List<Layer> layers = layerRepository.findByServiceIdIn(neededServiceIds);
327 
328         // Check the visibility of each visible StartLayer's corresponding Layer.
329         for (StartLayer startLayer : visibleStartLayers) {
330             ApplicationLayer applicationLayer = startLayer.getApplicationLayer();
331             Layer serviceVisibilityLayer = null;
332             for (Layer l : layers) {
333                 if (l.getService().equals(applicationLayer.getService())
334                         && Objects.equals(l.getName(), applicationLayer.getLayerName())) {
335                     serviceVisibilityLayer = l;
336                     break;
337                 }
338             }
339 
340             assert serviceVisibilityLayer != null;
341             boolean isLayerVisible =
342                     isAuthorized(serviceVisibilityLayer.getService().getReaders(), authentication);
343             while (isLayerVisible && serviceVisibilityLayer != null) {
344                 if (!isAuthorized(serviceVisibilityLayer.getReaders(), authentication)) {
345                     isLayerVisible = false;
346                     break;
347                 }
348 
349                 serviceVisibilityLayer = serviceVisibilityLayer.getParent();
350             }
351 
352             // If the Layer is not visible, remove the ApplicationLayer from the layerMap.
353             if (!isLayerVisible) {
354                 layerMap.remove(applicationLayer.getId());
355             }
356         }
357 
358         // Repopulate the visible StartLayer list while iterating over StartLevels, this time taking
359         // each Layer's visibility in account.
360         visibleStartLayers.clear();
361 
362         Map<Long, LayerTreeNode> treeNodeMap = new HashMap<>();
363         for (StartLevel l : startLevels) {
364             // Check if this level is a child of a background level. In the API background levels
365             // are returned in a separate tree. Only children of the Level with isBackground() set
366             // to true can be a StartLevel, so we need to check all parents only (not the Level of
367             // the StartLevel itself).
368             boolean isBackground = false;
369             Level parentLevel = l.getLevel();
370             while (parentLevel != null && !isBackground) {
371                 isBackground = parentLevel.isBackground();
372                 parentLevel = parentLevel.getParent();
373             }
374 
375             Level startLevel = l.getLevel();
376             List<LayerTreeNode> treeNodeList;
377             LayerTreeNode chosenRoot;
378             if (isBackground) {
379                 treeNodeList = mapResponse.getBaseLayerTreeNodes();
380                 chosenRoot = rootBackgroundNode;
381             } else {
382                 treeNodeList = mapResponse.getLayerTreeNodes();
383                 chosenRoot = rootNode;
384             }
385 
386             levelQueue.add(startLevel);
387             while (!levelQueue.isEmpty()) {
388                 Level level = levelQueue.pop();
389                 if (treeNodeMap.containsKey(level.getId())
390                         || !visibleLevels.contains(level.getId())) {
391                     continue;
392                 }
393 
394                 // Use a prefix to make the LayerTreeNode ids in the tree containing both Level and
395                 // ApplicationLayer nodes unique
396                 LayerTreeNode childNode =
397                         new LayerTreeNode()
398                                 .id(String.format("lvl_%d", level.getId()))
399                                 .name(level.getName())
400                                 .root(false)
401                                 .childrenIds(new ArrayList<>());
402 
403                 treeNodeList.add(childNode);
404                 treeNodeMap.put(level.getId(), childNode);
405 
406                 LayerTreeNode parentNode;
407                 if (level == startLevel) {
408                     parentNode = chosenRoot;
409                 } else {
410                     parentNode = treeNodeMap.get(level.getParent().getId());
411                 }
412                 parentNode.addChildrenIdsItem(childNode.getId());
413 
414                 levelQueue.addAll(levelChildrenMap.getOrDefault(level.getId(), List.of()));
415                 for (ApplicationLayer layer : level.getLayers()) {
416                     StartLayer startLayer = layerMap.get(layer.getId());
417                     if (startLayer == null) {
418                         continue;
419                     }
420 
421                     visibleStartLayers.add(startLayer);
422 
423                     LayerTreeNode layerNode =
424                             new LayerTreeNode()
425                                     .id(String.format("lyr_%d", layer.getId()))
426                                     .name(getNameForAppLayer(layer, layers))
427                                     .appLayerId((int) (long) layer.getId())
428                                     .root(false)
429                                     .childrenIds(new ArrayList<>());
430 
431                     treeNodeList.add(layerNode);
432                     childNode.addChildrenIdsItem(layerNode.getId());
433                 }
434             }
435         }
436 
437         // Only add ApplicationLayers visible in the LayerTreeNode graph to the response.
438         for (StartLayer l : visibleStartLayers) {
439             ApplicationLayer applicationLayer = l.getApplicationLayer();
440             Layer serviceLayer = null;
441             for (Layer layer : layers) {
442                 if (layer.getService().equals(applicationLayer.getService())
443                         && Objects.equals(layer.getName(), applicationLayer.getLayerName())) {
444                     serviceLayer = layer;
445                     break;
446                 }
447             }
448 
449             AppLayer.HiDpiModeEnum hiDpiMode = null;
450             String hiDpiSubstituteLayer = null;
451 
452             if (serviceLayer != null) {
453                 ClobElement ce = serviceLayer.getDetails().get("hidpi.mode");
454                 if (ce != null) {
455                     try {
456                         hiDpiMode = AppLayer.HiDpiModeEnum.fromValue(ce.getValue());
457                     } catch (IllegalArgumentException e) {
458                         logger.warn(
459                                 String.format(
460                                         "App #%s (%s): invalid hidpi.mode enum value for app layer #%s, service layer #%s (%s)",
461                                         a.getId(),
462                                         a.getNameWithVersion(),
463                                         l.getId(),
464                                         serviceLayer.getId(),
465                                         serviceLayer.getName()));
466                     }
467                 }
468                 ce = serviceLayer.getDetails().get("hidpi.substitute_layer");
469                 if (ce != null) {
470                     hiDpiSubstituteLayer = ce.getValue();
471                 }
472             }
473 
474             assert serviceLayer != null;
475             AppLayer appLayer =
476                     new AppLayer()
477                             .id(applicationLayer.getId())
478                             .layerName(serviceLayer.getName())
479                             .title(getNameForAppLayer(applicationLayer, layers))
480                             .serviceId(applicationLayer.getService().getId())
481                             .visible(l.isChecked())
482                             .maxScale(serviceLayer.getMaxScale())
483                             .minScale(serviceLayer.getMinScale())
484                             .legendImageUrl(serviceLayer.getLegendImageUrl())
485                             .hiDpiMode(hiDpiMode)
486                             .hiDpiSubstituteLayer(hiDpiSubstituteLayer)
487                             .hasAttributes(!l.getApplicationLayer().getAttributes().isEmpty());
488 
489             mapResponse.addAppLayersItem(appLayer);
490 
491             GeoService geoService = applicationLayer.getService();
492 
493             // Use this default if saved before the form default was added in admin
494             Service.ServerTypeEnum serviceServerType = Service.ServerTypeEnum.AUTO;
495             ClobElement ce = geoService.getDetails().get("serverType");
496             if (ce != null) {
497                 try {
498                     serviceServerType = Service.ServerTypeEnum.fromValue(ce.getValue());
499                 } catch (IllegalArgumentException e) {
500                     logger.warn(
501                             String.format(
502                                     "App #%s (%s): invalid serverType enum value for service #%s (%s)",
503                                     a.getId(),
504                                     a.getNameWithVersion(),
505                                     geoService.getId(),
506                                     geoService.getName()));
507                 }
508             }
509             Integer tilingGutter = null;
510             ce = geoService.getDetails().get("tiling.gutter");
511             if (ce != null) {
512                 try {
513                     tilingGutter = Integer.parseInt(ce.getValue());
514                 } catch (NumberFormatException ignored) {
515                     // ignored
516                 }
517             }
518 
519             Service s =
520                     new Service()
521                             .url(geoService.getUrl())
522                             .id(geoService.getId())
523                             .name(geoService.getName())
524                             .protocol(Service.ProtocolEnum.fromValue(geoService.getProtocol()))
525                             .serverType(serviceServerType)
526                             .tilingDisabled(
527                                     "true"
528                                             .equals(
529                                                     geoService
530                                                             .getDetails()
531                                                             .getOrDefault(
532                                                                     "tiling.disable",
533                                                                     new ClobElement("false)"))
534                                                             .getValue()))
535                             .tilingGutter(tilingGutter)
536                             .capabilities(geoService.getCapabilitiesDoc());
537             if (geoService.getProtocol().equalsIgnoreCase(TileService.PROTOCOL)) {
538                 s.tilingProtocol(
539                         Service.TilingProtocolEnum.fromValue(
540                                 ((TileService) geoService).getTilingProtocol()));
541             }
542             mapResponse.addServicesItem(s);
543         }
544     }
