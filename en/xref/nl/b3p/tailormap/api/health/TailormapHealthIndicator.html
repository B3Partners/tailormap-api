6   package nl.b3p.tailormap.api.health;
7   
8   import nl.b3p.tailormap.api.controller.VersionController;
9   
10  import org.apache.commons.logging.Log;
11  import org.apache.commons.logging.LogFactory;
12  import org.springframework.beans.factory.annotation.Value;
13  import org.springframework.boot.actuate.autoconfigure.health.ConditionalOnEnabledHealthIndicator;
14  import org.springframework.boot.actuate.health.Health;
15  import org.springframework.boot.actuate.health.HealthIndicator;
16  import org.springframework.stereotype.Component;
17  
18  /**
19   * HealthIndicator for Tailormap. Can be used to retrieve Tailormap specific health statistics from
20   * the actuator endpoint.
21   *
22   * @since 0.1
23   * @author mprins
24   */
25  @Component
26  @ConditionalOnEnabledHealthIndicator("tailormap")
27  public class TailormapHealthIndicator implements HealthIndicator {
28      private final Log logger = LogFactory.getLog(getClass());
29      private final VersionController versionController;
30  
31      @Value("${management.health.tailormap.enabled}")
32      private boolean healthEnabled = false;
33  
34      public TailormapHealthIndicator(VersionController versionController) {
35          this.versionController = versionController;
36      }
37  
38      @Override
39      public Health health() {
40          Health.Builder health = Health.unknown();
41  
42          logger.info("tailormap health indicator " + (healthEnabled ? "enabled" : "disabled"));
43  
44          if (healthEnabled) {
45              try {
46                  // TODO check some other statistics
47                  health.up().withDetails(versionController.getVersion());
48              } catch (Exception e) {
49                  logger.fatal("Error checking Tailormap health. " + e.getMessage());
50                  logger.debug("Error checking Tailormap health", e);
51                  health.outOfService().withDetail("message", "Error checking database");
52              }
53          }
54  
55          return health.build();
56      }
57  }
