10  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;
11  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.jsonPath;
12  import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
13  
14  import nl.b3p.tailormap.api.HSQLDBTestProfileJPAConfiguration;
15  import nl.b3p.tailormap.api.controller.VersionController;
16  
17  import org.junit.jupiter.api.Test;
18  import org.springframework.beans.factory.annotation.Autowired;
19  import org.springframework.boot.autoconfigure.EnableAutoConfiguration;
20  import org.springframework.boot.test.autoconfigure.actuate.metrics.AutoConfigureMetrics;
21  import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
22  import org.springframework.boot.test.context.SpringBootTest;
23  import org.springframework.http.MediaType;
24  import org.springframework.test.context.ActiveProfiles;
25  import org.springframework.test.context.TestPropertySource;
26  import org.springframework.test.web.servlet.MockMvc;
27  
28  @SpringBootTest(
29          classes = {
30              HSQLDBTestProfileJPAConfiguration.class,
31              TailormapHealthIndicator.class,
32              VersionController.class
33          })
34  @AutoConfigureMockMvc
35  @EnableAutoConfiguration
36  @TestPropertySource(properties = {"management.health.tailormap.enabled=true"})
37  @ActiveProfiles("test")
38  @AutoConfigureMetrics
39  class TailormapHealthIndicatorEnabledIntegrationTest {
40      @Autowired private MockMvc mockMvc;
41  
42      @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
43      @Test
44      void when_enabled_health_should_have_status_and_response_data() throws Exception {
45          String projectVersion = System.getProperty("project.version");
46          assumeFalse(
47                  null == projectVersion,
48                  "Project version unknown, should be set as system property");
49          String databaseVersion = System.getenv("DATABASE_VERSION");
50          assumeFalse(
51                  null == databaseVersion,
52                  "Database version unknown, should be set in system environment");
53          String apiVersion = System.getenv("API_VERSION");
54          assumeFalse(null == apiVersion, "API version unknown, should be set in system environment");
55  
56          mockMvc.perform(get("/actuator/health/tailormap"))
57                  .andExpect(status().isOk())
58                  .andExpect(
59                          content()
60                                  .contentType(
61                                          MediaType.parseMediaType(
62                                                  "application/vnd.spring-boot.actuator.v3+json")))
63                  .andExpect(jsonPath("$.details.version").value(projectVersion))
64                  .andExpect(jsonPath("$.details.apiVersion").value(apiVersion))
65                  .andExpect(jsonPath("$.details.databaseversion").value(databaseVersion))
66                  .andExpect(jsonPath("$.details.buildDate").isNotEmpty());
67      }
68  }
