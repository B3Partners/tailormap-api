58  /**
59   * Proxy controller for OGC WMS and WMTS services. Does not attempt to hide the original service
60   * URL. Mostly useful for access to HTTP Basic secured services without sending the credentials to
61   * the client. The access control is handled by Spring Security and the authorizations configured on
62   * the service.
63   *
64   * <p>Only supports GET requests. Does not support CORS, only meant for tailormap-viewer from the
65   * same origin.
66   *
67   * <p>Implementation note: uses the Java 11 HttpClient. Spring cloud gateway can proxy with many
68   * more features but can not be used in a non-reactive application.
69   */
70  @RestController
71  @Validated
72  @RequestMapping(path = "/app/{appId}/layer/{appLayerId}/proxy/{protocol}")
73  public class GeoServiceProxyController {
74  
75      private static final Log logger = LogFactory.getLog(GeoServiceProxyController.class);
76      private final ApplicationRepository applicationRepository;
77      private final ApplicationLayerRepository applicationLayerRepository;
78      private final AuthorizationService authorizationService;
79  
80      @Autowired
81      public GeoServiceProxyController(
82              ApplicationRepository applicationRepository,
83              ApplicationLayerRepository applicationLayerRepository,
84              AuthorizationService authorizationService) {
85          this.applicationRepository = applicationRepository;
86          this.applicationLayerRepository = applicationLayerRepository;
87          this.authorizationService = authorizationService;
88      }
89  
90      @RequestMapping(method = {GET, POST})
91      public ResponseEntity<?> proxy(
92              @Parameter(name = "appId", description = "application id", required = true)
93                      @PathVariable("appId")
94                      Long appId,
95              @Parameter(name = "appLayerId", description = "application layer id", required = true)
96                      @PathVariable("appLayerId")
97                      Long appLayerId,
98              @PathVariable("protocol") String protocol,
99              HttpServletRequest request) {
100 
101         Predicate<GeoService> serviceValidator;
102         if ("wmts".equals(protocol)) {
103             serviceValidator =
104                     (GeoService gs) ->
105                             TileService.PROTOCOL.equals(gs.getProtocol())
106                                     && TileService.TILING_PROTOCOL_WMTS.equals(
107                                             ((TileService) gs).getTilingProtocol());
108         } else if ("wms".equals(protocol)) {
109             serviceValidator = (GeoService gs) -> WMSService.PROTOCOL.equals(gs.getProtocol());
110         } else {
111             return ResponseEntity.status(HttpStatus.NOT_FOUND)
112                     .body("Unknown proxy protocol: " + protocol);
113         }
114 
115         final Application application = applicationRepository.findById(appId).orElse(null);
116         final ApplicationLayer appLayer =
117                 applicationLayerRepository.findById(appLayerId).orElse(null);
118         if (application == null || appLayer == null) {
119             return ResponseEntity.status(HttpStatus.NOT_FOUND)
120                     .body("Requested an application or appLayer that does not exist");
121         }
122 
123         if (!authorizationService.mayUserRead(appLayer, application)) {
124             // login required, send RedirectResponse
125             return ResponseEntity.status(HttpStatus.UNAUTHORIZED).body(new RedirectResponse());
126         } else {
127             final GeoService service = appLayer.getService();
128 
129             if (!serviceValidator.test(service)) {
130                 return ResponseEntity.status(HttpStatus.BAD_REQUEST)
131                         .body("Invalid proxy protocol for service");
132             }
133 
134             if (!Boolean.parseBoolean(
135                     String.valueOf(service.getDetails().get(GeoService.DETAIL_USE_PROXY)))) {
136                 return ResponseEntity.status(HttpStatus.FORBIDDEN)
137                         .body("Proxy not enabled for requested service");
138             }
139 
140             final UriComponentsBuilder originalServiceUrl =
141                     UriComponentsBuilder.fromHttpUrl(service.getUrl());
142             // request.getParameterMap() includes parameters from an
143             // application/x-www-form-urlencoded POST body
144             final MultiValueMap<String, String> requestParams =
145                     request.getParameterMap().entrySet().stream()
146                             .map(
147                                     entry ->
148                                             new AbstractMap.SimpleEntry<>(
149                                                     entry.getKey(),
150                                                     Arrays.stream(entry.getValue())
151                                                             .map(
152                                                                     value ->
153                                                                             UriUtils.encode(
154                                                                                     value,
155                                                                                     StandardCharsets
156                                                                                             .UTF_8))
157                                                             .collect(Collectors.toList())))
158                             .collect(
159                                     Collectors.toMap(
160                                             Map.Entry::getKey,
161                                             Map.Entry::getValue,
162                                             (x, y) -> y,
163                                             LinkedMultiValueMap::new));
164             final MultiValueMap<String, String> params =
165                     buildOgcProxyRequestParams(
166                             originalServiceUrl.build(true).getQueryParams(), requestParams);
167             originalServiceUrl.replaceQueryParams(params);
168 
169             return doProxy(
170                     originalServiceUrl.build(true).toUri(),
171                     application,
172                     appLayer,
173                     service,
174                     request);
175         }
176     }
177 
178     public static MultiValueMap<String, String> buildOgcProxyRequestParams(
179             MultiValueMap<String, String> originalServiceParams,
180             MultiValueMap<String, String> requestParams) {
181         // Start with all the parameters from the request
182         final MultiValueMap<String, String> params = new LinkedMultiValueMap<>(requestParams);
183 
184         // Add original service URL parameters if they are not required OGC params
185         // (case-insensitive) and not already set by request
186         final List<String> ogcParams = List.of(new String[] {"SERVICE", "REQUEST", "VERSION"});
187         for (Map.Entry<String, List<String>> serviceParam : originalServiceParams.entrySet()) {
188             if (!params.containsKey(serviceParam.getKey())
189                     && !ogcParams.contains(serviceParam.getKey().toUpperCase())) {
190                 params.put(serviceParam.getKey(), serviceParam.getValue());
191             }
192         }
193         return params;
194     }

196     private static ResponseEntity<?> doProxy(
197             URI uri,
198             Application application,
199             ApplicationLayer appLayer,
200             GeoService service,
201             HttpServletRequest request) {
202         final HttpClient.Builder builder =
203                 HttpClient.newBuilder().followRedirects(HttpClient.Redirect.NORMAL);
204 
205         // Some HTTP services may not like HTTP/2 (not configurable in admin at the moment)
206         if (Boolean.parseBoolean(String.valueOf(service.getDetails().get("proxy_http1_1")))) {
207             builder.version(HttpClient.Version.HTTP_1_1);
208         }
209 
210         final HttpClient httpClient = builder.build();
211 
212         HttpRequest.Builder requestBuilder = HttpRequest.newBuilder().uri(uri);
213 
214         try {
215             String ip = request.getRemoteAddr();
216             InetAddress inetAddress = InetAddress.getByName(ip);
217 
218             if (inetAddress instanceof Inet6Address) {
219                 // https://stackoverflow.com/questions/33168783/
220                 Inet6Address inet6Address = (Inet6Address) inetAddress;
221                 int scopeId = inet6Address.getScopeId();
222 
223                 if (scopeId > 0) {
224                     ip = inet6Address.getHostName().replaceAll("%" + scopeId + "$", "");
225                 }
226 
227                 // IPv6 address must be bracketed and quoted
228                 ip = "\"[" + ip + "]\"";
229             }
230             requestBuilder.header("X-Forwarded-For", ip);
231             requestBuilder.header("Forwarded", "for=" + ip);
232         } catch (UnknownHostException ignored) {
233             // Don't care
234         }
235 
236         String[] requestHeaders = {
237             "Accept",
238             "If-Modified-Since",
239             "If-Unmodified-Since",
240             "If-Match",
241             "If-None-Match",
242             "If-Range",
243             "Range",
244             "Referer",
245             "User-Agent",
246         };
247         for (String header : requestHeaders) {
248             String value = request.getHeader(header);
249             if (value != null) {
250                 requestBuilder.header(header, value);
251             }
252         }
253 
254         boolean addCredentials = service.getUsername() != null && service.getPassword() != null;
255         if (addCredentials) {
256             // An app admin should not be able to accidentally open a proxy to a secured geo service
257             // in a public app, so only allow proxying to a secured geo service in apps which
258             // require
259             // authentication.
260 
261             // A geo service which is in a private network which does not require credentials /can/
262             // be 'exposed' in a public app through this proxy.
263 
264             // An app may have authenticated required but "no groups checked" to allow anyone to
265             // access the service (after being logged in for the app). This might still be an
266             // inadvertent misconfiguration but less concerning.
267 
268             // Previous tailormap proxy behaviour was to proxy the request but not add the
269             // authentication. This proxy sends a Forbidden response.
270 
271             if (!application.isAuthenticatedRequired()) {
272                 // XXX for tiled services each request logs a warning, may clutter the log...
273                 logger.warn(
274                         String.format(
275                                 "App %s has app layer %s from proxied secured service URL %s (username %s), but app authentication is not required. Denying proxy, even if user is authenticated.",
276                                 application.getId(),
277                                 appLayer.getId(),
278                                 service.getUrl(),
279                                 service.getUsername()));
280                 return ResponseEntity.status(HttpStatus.FORBIDDEN).body("Forbidden");
281             }
282             String toEncode = service.getUsername() + ":" + service.getPassword();
283             requestBuilder.header(
284                     "Authorization",
285                     "Basic "
286                             + Base64.getEncoder()
287                                     .encodeToString(toEncode.getBytes(StandardCharsets.UTF_8)));
288 
289             // TODO: in some cases we might want to forward the authenticated tailormap user or
290             // groups/roles to the service (perhaps in a secret header name) so the service can
291             // apply different authorizations instead of just one account.
292         }
293 
294         try {
295             HttpResponse<InputStream> response =
296                     httpClient.send(
297                             requestBuilder.build(), HttpResponse.BodyHandlers.ofInputStream());
298             // If the server does not accept our credentials it might 'hide' the layer or even send
299             // a 401 Unauthorized status. We do not send the WWW-Authenticate header back, so the
300             // client will get the error but not an authorization popup.
301             // It would be nice if proxy (auth) errors were logged and available in the admin
302             // interface. Currently, a layer will just stop working if the geo service credentials
303             // are changed without updating them in the geo service registry.
304             InputStreamResource body = new InputStreamResource(response.body());
305             org.springframework.http.HttpHeaders headers = new HttpHeaders();
306             // TODO for WMTS, does caching work?
307             String[] allowedResponseHeaders = {
308                 "Content-Type",
309                 "Content-Length",
310                 "Content-Range",
311                 "Content-Disposition",
312                 "Cache-Control",
313                 "Expires",
314                 "Last-Modified",
315                 "ETag",
316                 "Pragma",
317             };
318             for (String header : allowedResponseHeaders) {
319                 headers.addAll(header, response.headers().allValues(header));
320             }
321             return ResponseEntity.status(response.statusCode()).headers(headers).body(body);
322         } catch (Exception e) {
323             return ResponseEntity.status(HttpStatus.BAD_GATEWAY).body("Bad Gateway");
324         }
325     }
