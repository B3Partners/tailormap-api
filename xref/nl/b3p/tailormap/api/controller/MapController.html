45  @RestController
46  @Validated
47  @RequestMapping(path = "/map/{appId}", produces = MediaType.APPLICATION_JSON_VALUE)
48  public class MapController {
49      private final Log logger = LogFactory.getLog(getClass());
50      @Autowired private ApplicationRepository applicationRepository;
51  
52      @Operation(
53              summary = "",
54              tags = {},
55              responses = {
56                  @ApiResponse(
57                          responseCode = "200",
58                          description = "OK",
59                          content =
60                                  @Content(
61                                          mediaType = MediaType.APPLICATION_JSON_VALUE,
62                                          schema = @Schema(implementation = MapResponse.class))),
63                  @ApiResponse(
64                          responseCode = "400",
65                          description = "Bad Request",
66                          content =
67                                  @Content(
68                                          mediaType = MediaType.APPLICATION_JSON_VALUE,
69                                          schema = @Schema(implementation = ErrorResponse.class)))
70              })
71      @GetMapping(produces = MediaType.APPLICATION_JSON_VALUE)
72      @ResponseBody
73      public MapResponse get(
74              @Parameter(name = "appId", description = "application id", required = true)
75                      @PathVariable("appId")
76                      Long appId) {
77          logger.trace("Requesting 'map' for application id: " + appId);
78  
79          MapResponse mapResponse = new MapResponse();
80  
81          // this could throw EntityNotFound, which is handles by handleEntityNotFoundException
82          // and in a normal flow this should not happen
83          // as appId is (should be) validated by calling the /app/ endpoint
84          Application application = applicationRepository.getById(appId);
85  
86          getApplicationParams(application, mapResponse);
87          getLayers(application, mapResponse);
88  
89          return mapResponse;
90      }
91  
92      @ExceptionHandler(EntityNotFoundException.class)
93      @ResponseStatus(
94              value =
95                      HttpStatus
96                              .BAD_REQUEST /*,reason = "Bad Request" -- adding 'reason' will drop the body */)
97      @ResponseBody
98      public ErrorResponse handleEntityNotFoundException(EntityNotFoundException exception) {
99          logger.warn(
100                 "Requested an application that does not exist. Message: " + exception.getMessage());
101         return new ErrorResponse()
102                 .message("Requested an application that does not exist")
103                 .code(400);
104     }
105 
106     private void getApplicationParams(@NotNull Application a, @NotNull MapResponse mapResponse) {
107         final String pCode = a.getProjectionCode();
108         //        final String pCode =
109         //                "EPSG:28992[+proj=sterea +lat_0=52.15616055555555 +lon_0=5.38763888888889
110         // +k=0.9999079 +x_0=155000 +y_0=463000 +ellps=bessel
111         // +towgs84=565.237,50.0087,465.658,-0.406857,0.350733,-1.87035,4.0812 +units=m +no_defs]";
112         CoordinateReferenceSystem c = new CoordinateReferenceSystem();
113         if (null != pCode) {
114             c.code(pCode.substring(0, pCode.indexOf('[')))
115                     .definition(pCode.substring(pCode.indexOf('[') + 1, pCode.lastIndexOf(']')));
116         }
117         Bounds maxExtent = new Bounds();
118         if (null != a.getMaxExtent()) {
119             maxExtent
120                     .minx(a.getMaxExtent().getMinx())
121                     .miny(a.getMaxExtent().getMiny())
122                     .maxx(a.getMaxExtent().getMaxx())
123                     .maxy(a.getMaxExtent().getMaxy())
124                     .crs(a.getMaxExtent().getCrs().getName());
125         }
126         Bounds initialExtent = new Bounds();
127         if (null != a.getStartExtent()) {
128             initialExtent
129                     .minx(a.getStartExtent().getMinx())
130                     .miny(a.getStartExtent().getMiny())
131                     .maxx(a.getStartExtent().getMaxx())
132                     .maxy(a.getStartExtent().getMaxy())
133                     .crs(a.getStartExtent().getCrs().getName());
134         }
135 
136         mapResponse.crs(c).maxExtent(maxExtent).initialExtent(initialExtent);
137     }
138 
139     private void getLayers(@NotNull Application a, @NotNull MapResponse mapResponse) {
140         List<StartLayer> startLayers = a.getStartLayers();
141         if (null != a.getStartLayers()) {
142             for (StartLayer l : startLayers) {
143                 AppLayer appLayer =
144                         new AppLayer()
145                                 .id(l.getApplicationLayer().getId())
146                                 .displayName(l.getApplicationLayer().getLayerName())
147                                 .url(l.getApplicationLayer().getService().getUrl())
148                                 .serviceId(l.getApplicationLayer().getService().getId())
149                                 // TODO fixup hardcoded data
150                                 .crs(new CoordinateReferenceSystem())
151                                 .visible(true)
152                                 .isBaseLayer(true);
153 
154                 GeoService geoService = l.getApplicationLayer().getService();
155                 Service s =
156                         new Service()
157                                 .url(geoService.getUrl())
158                                 .id(geoService.getId())
159                                 .name(geoService.getName())
160                                 .protocol(Service.ProtocolEnum.fromValue(geoService.getProtocol()))
161                                 .capabilities(geoService.getCapabilitiesDoc());
162                 if (geoService.getProtocol().equalsIgnoreCase(TileService.PROTOCOL)) {
163                     s.tilingProtocol(
164                             Service.TilingProtocolEnum.fromValue(
165                                     ((TileService) geoService).getTilingProtocol()));
166                 }
167                 if (appLayer.getIsBaseLayer()) {
168                     mapResponse.addBaseLayersItem(appLayer);
169                 }
170                 mapResponse.addServicesItem(s);
171             }
172         }
173     }
