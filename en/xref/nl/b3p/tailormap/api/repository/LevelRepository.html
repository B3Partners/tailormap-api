6   package nl.b3p.tailormap.api.repository;
7   
8   import nl.tailormap.viewer.config.app.Level;
9   
10  import org.springframework.data.jpa.repository.EntityGraph;
11  import org.springframework.data.jpa.repository.JpaRepository;
12  import org.springframework.data.jpa.repository.Query;
13  
14  import java.util.Collection;
15  import java.util.List;
16  
17  /**
18   * Easy to use methods to access {@link Level}.
19   *
20   * @since 0.1
21   */
22  public interface LevelRepository extends JpaRepository<Level, Long> {
23      @Query(
24              value =
25                      "select * from level_ where id in (with recursive level_tree(id) as (select id from level_ where id = ?1 union all select l.id from level_tree, level_ l where l.parent = level_tree.id) select * from level_tree)",
26              nativeQuery = true)
27      List<Level> findByLevelTree(Long rootId);
28  
29      @EntityGraph(attributePaths = {"readers", "layers.service", "layers.readers"})
30      List<Level> findWithAuthorizationDataByIdIn(Collection<Long> id);
31  }
