6   package nl.b3p.tailormap.api.security;
7   
8   import nl.b3p.tailormap.api.persistence.Group;
9   import org.apache.commons.lang3.ArrayUtils;
10  import org.springframework.beans.factory.annotation.Value;
11  import org.springframework.context.EnvironmentAware;
12  import org.springframework.context.annotation.Bean;
13  import org.springframework.context.annotation.Configuration;
14  import org.springframework.core.env.Environment;
15  import org.springframework.http.HttpStatus;
16  import org.springframework.security.config.annotation.method.configuration.EnableMethodSecurity;
17  import org.springframework.security.config.annotation.web.builders.HttpSecurity;
18  import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
19  import org.springframework.security.web.SecurityFilterChain;
20  import org.springframework.security.web.csrf.CookieCsrfTokenRepository;
21  
22  @Configuration
23  @EnableWebSecurity
24  @EnableMethodSecurity
25  public class ApiSecurityConfiguration implements EnvironmentAware {
26    @Value("${tailormap-api.base-path}")
27    private String apiBasePath;
28  
29    @Value("${tailormap-api.admin.base-path}")
30    private String adminApiBasePath;
31  
32    private Environment environment;
33  
34    @Override
35    public void setEnvironment(Environment environment) {
36      this.environment = environment;
37    }
38  
39    @Bean
40    public CookieCsrfTokenRepository csrfTokenRepository() {
41      // Note: CSRF protection only required when using cookies for authentication. This requires an
42      // X-XSRF-TOKEN header read from the XSRF-TOKEN cookie by JavaScript so set HttpOnly to false.
43      // Angular has automatic XSRF protection support:
44      // https://angular.io/guide/http#security-xsrf-protection
45      CookieCsrfTokenRepository csrfTokenRepository = CookieCsrfTokenRepository.withHttpOnlyFalse();
46      csrfTokenRepository.setCookiePath("/");
47      return csrfTokenRepository;
48    }
49  
50    @Bean
51    public SecurityFilterChain apiFilterChain(
52        HttpSecurity http, CookieCsrfTokenRepository csrfTokenRepository) throws Exception {
53  
54      // Disable CSRF protection for development with HAL explorer
55      // https://github.com/spring-projects/spring-data-rest/issues/1347
56      if (ArrayUtils.contains(environment.getActiveProfiles(), "disable-csrf")) {
57        http.csrf().disable();
58      } else {
59        http = http.csrf().csrfTokenRepository(csrfTokenRepository).and();
60      }
61  
62      http.securityMatchers(matchers -> matchers.requestMatchers(apiBasePath + "/**"))
63          .authorizeHttpRequests(
64              authorize ->
65                  authorize.requestMatchers(adminApiBasePath + "/**").hasAuthority(Group.ADMIN))
66          .formLogin()
67          .loginProcessingUrl(apiBasePath + "/login")
68          .and()
69          .logout()
70          .logoutUrl(apiBasePath + "/logout")
71          .logoutSuccessHandler(
72              (request, response, authentication) -> response.sendError(HttpStatus.OK.value(), "OK"));
73      return http.build();
74    }
75  }
