6   package nl.b3p.tailormap.api.configuration.base;
7   
8   import java.util.List;
9   import java.util.Locale;
10  import java.util.Objects;
11  import javax.servlet.http.HttpServletRequest;
12  import org.springframework.stereotype.Controller;
13  import org.springframework.web.bind.annotation.GetMapping;
14  import org.springframework.web.bind.annotation.PathVariable;
15  import org.springframework.web.servlet.i18n.AcceptHeaderLocaleResolver;
16  
17  /**
18   * Front controller which forwards requests to URLs created by the Angular routing module to the
19   * proper viewer or admin index.html of the Angular bundle, for example when the user refreshes the
20   * page. Unfortunately, a proper front controller would first check if the file which the requested
21   * URL maps to exists and only return the index.html file if it doesn't. That is quite complicated
22   * to do in Spring Boot so as a poor-mans alternative we explicitly match the specific router URLs
23   * navigated to by the Angular frontends which don't map to files from the bundle.
24   *
25   * <p>It could happen that a frontend developer adds a route and forgets to update this front
26   * controller, and it is only noticed later when someone presses F5 and gets a 404.
27   */
28  @Controller
29  public class FrontController {
30    // Hardcoded list for now. In the future scan the spring.web.resources.static-locations directory
31    // for subdirectories of locale-specific frontend bundles.
32    private static final AcceptHeaderLocaleResolver localeResolver = new AcceptHeaderLocaleResolver();
33    private static final String DEFAULT_LOCALE;
34  
35    static {
36      localeResolver.setSupportedLocales(List.of(new Locale("en"), new Locale("nl")));
37      localeResolver.setDefaultLocale(localeResolver.getSupportedLocales().get(0));
38      DEFAULT_LOCALE = Objects.requireNonNull(localeResolver.getDefaultLocale()).toLanguageTag();
39    }
40  
41    @GetMapping(value = {"/", "/login", "/app/**", "/service/**", "/admin/**"})
42    public String appIndex(HttpServletRequest request) {
43      Locale locale = localeResolver.resolveLocale(request);
44      String path = "/" + locale.toLanguageTag() + request.getRequestURI();
45      String query = request.getQueryString() == null ? "" : "?" + request.getQueryString();
46      return "redirect:" + path + query;
47    }
48  
49    @GetMapping(
50        value = {
51          "/{locale}/",
52          "/{locale}/login",
53          // Need to avoid matching /api/app/
54          "/{locale:^(?!api)[a-zA-Z-]+}/app/**",
55          "/{locale:^(?!api)[a-zA-Z-]+}/service/**",
56          "/{locale:^(?!api)[a-zA-Z-]+}/admin/**"
57        })
58    public String appRoutes(@PathVariable("locale") String locale) {
59      if (localeResolver.getSupportedLocales().stream()
60          .anyMatch(l -> l.toLanguageTag().equals(locale))) {
61        return "forward:/" + locale + "/index.html";
62      }
63      return "forward:/" + DEFAULT_LOCALE + "/index.html";
64    }
65  
66    @GetMapping(value = {"/swagger-ui", "/swagger-ui/"})
67    public String swaggerUiWelcomePage() {
68      return "redirect:/swagger-ui/index.html";
69    }
70  }
