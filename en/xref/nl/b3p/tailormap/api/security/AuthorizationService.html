6   package nl.b3p.tailormap.api.security;
7   
8   import nl.b3p.tailormap.api.repository.LayerRepository;
9   import nl.b3p.tailormap.api.repository.LevelRepository;
10  import nl.tailormap.viewer.config.app.Application;
11  import nl.tailormap.viewer.config.app.ApplicationLayer;
12  import nl.tailormap.viewer.config.app.Level;
13  import nl.tailormap.viewer.config.services.GeoService;
14  import nl.tailormap.viewer.config.services.Layer;
15  
16  import org.springframework.security.authentication.AnonymousAuthenticationToken;
17  import org.springframework.security.core.Authentication;
18  import org.springframework.security.core.context.SecurityContextHolder;
19  import org.springframework.stereotype.Service;
20  
21  import java.util.HashSet;
22  import java.util.Set;
23  
24  /**
25   * Validates access control rules. Any call to mayUserRead will verify that the currently logged in
26   * user is not only allowed to read the current object, but any objcet above and below it in the
27   * hierarchy.
28   */
29  @Service
30  public class AuthorizationService {
31      private final LevelRepository levelRepository;
32      private final LayerRepository layerRepository;
33  
34      public AuthorizationService(LevelRepository levelRepository, LayerRepository layerRepository) {
35          this.levelRepository = levelRepository;
36          this.layerRepository = layerRepository;
37      }
38  
39      /**
40       * Verifies that the user may read the object, based on the passed in readers set.
41       *
42       * @param readers the list of readers to validate against.
43       * @return the result of validating the authorizations.
44       */
45      private boolean isAuthorizedBySet(Set<String> readers) {
46          if (readers == null || readers.isEmpty()) {
47              return true;
48          }
49  
50          Authentication auth = SecurityContextHolder.getContext().getAuthentication();
51          if (auth == null) {
52              return false;
53          }
54  
55          for (String reader : readers) {
56              if (auth.getAuthorities().stream().anyMatch(x -> x.getAuthority().equals(reader))) {
57                  return true;
58              }
59          }
60  
61          return false;
62      }
63  
64      /**
65       * Verifies that this user may read this Application.
66       *
67       * @param application the Application to check
68       * @return the results from the access control checks.
69       */
70      public boolean mayUserRead(Application application) {
71          Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
72          if ((authentication == null || authentication instanceof AnonymousAuthenticationToken)
73                  && application.isAuthenticatedRequired()) {
74              return false;
75          }
76  
77          return isAuthorizedBySet(application.getReaders());
78      }
79  
80      /**
81       * Verifies that this user may view an ApplicationLayer in the context of a certain Application.
82       * As an ApplicationLayer cannot be traced back to a single Application (e.g. mashups), this
83       * requires passing in the Application to use as context. The ApplicationLayer, its parent
84       * Levels, and the Application are validated, as are the Layer, all its parents, and the
85       * GeoService.
86       *
87       * @param applicationLayer the ApplicationLayer to check
88       * @param context the Application to verify this ApplicationLayer against
89       * @return the results from the access control checks.
90       */
91      public boolean mayUserRead(ApplicationLayer applicationLayer, Application context) {
92          if (!isAuthorizedBySet(applicationLayer.getReaders())) {
93              return false;
94          }
95  
96          if (!mayUserRead(context)) {
97              return false;
98          }
99  
100         Set<Long> levelIds = new HashSet<>();
101         for (Level level : levelRepository.findByLevelTree(context.getRoot().getId())) {
102             levelIds.add(level.getId());
103         }
104 
105         Level closestLevel = null;
106         for (Level level : levelRepository.findWithAuthorizationDataByIdIn(levelIds)) {
107             if (level.getLayers().contains(applicationLayer)) {
108                 closestLevel = level;
109                 break;
110             }
111         }
112 
113         if (closestLevel == null) {
114             // layer ID does not exist in this application.
115             return false;
116         }
117 
118         if (!mayUserRead(closestLevel)) {
119             return false;
120         }
121 
122         GeoService geoService = applicationLayer.getService();
123         if (geoService == null) {
124             return true;
125         }
126 
127         if (this.isProxiedSecuredServiceLayerInPublicApplication(context, applicationLayer)) {
128             return false;
129         }
130 
131         return mayUserRead(
132                 layerRepository.getByServiceAndName(geoService, applicationLayer.getLayerName()));
133     }
134 
135     /**
136      * When a service is proxied with a username and password, authentication must be required for
137      * the application otherwise access to the layer should be denied to prevent an app admin
138      * accidentally publishing private data from a secured service in a public application. This
139      * method checks whether this is the case for a certain ApplicationLayer in the context of an
140      * Application.
141      *
142      * @param application the application (can be a mashup)
143      * @param applicationLayer the application layer (may belong to a parent application)
144      * @return see above
145      */
146     public boolean isProxiedSecuredServiceLayerInPublicApplication(
147             Application application, ApplicationLayer applicationLayer) {
148         GeoService geoService = applicationLayer.getService();
149         if (geoService == null) {
150             return false;
151         }
152         if (Boolean.parseBoolean(
153                 String.valueOf(geoService.getDetails().get(GeoService.DETAIL_USE_PROXY)))) {
154             boolean isSecuredService =
155                     geoService.getUsername() != null && geoService.getPassword() != null;
156             return isSecuredService && !application.isAuthenticatedRequired();
157         } else {
158             return false;
159         }
160     }
161 
162     /**
163      * Verifies that this user may view a Layer. The Layer, its parents, and the GeoService are all
164      * validated.
165      *
166      * @param layer the Layer to check
167      * @return the results from the access control checks.
168      */
169     public boolean mayUserRead(Layer layer) {
170         if (!isAuthorizedBySet(layer.getReaders())) {
171             return false;
172         }
173 
174         if (!mayUserRead(layer.getService())) {
175             return false;
176         }
177 
178         if (layer.getParent() != null) {
179             return mayUserRead(layer.getParent());
180         }
181 
182         return true;
183     }
184 
185     /**
186      * Verifies that this user may view a Level. The Level and its parents are all validated.
187      *
188      * @param level the Level to check
189      * @return the results from the access control checks.
190      */
191     public boolean mayUserRead(Level level) {
192         if (!isAuthorizedBySet(level.getReaders())) {
193             return false;
194         }
195 
196         if (level.getParent() != null) {
197             return mayUserRead(level.getParent());
198         }
199 
200         return true;
201     }
202 
203     /**
204      * Verifies that this user may view a GeoService. The GeoService is validated.
205      *
206      * @param geoService the GeoService to check
207      * @return the results from the access control checks.
208      */
209     public boolean mayUserRead(GeoService geoService) {
210         return isAuthorizedBySet(geoService.getReaders());
211     }
212 }
