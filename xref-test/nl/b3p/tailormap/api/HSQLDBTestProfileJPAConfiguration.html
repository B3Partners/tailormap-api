6   package nl.b3p.tailormap.api;
7   
8   import org.apache.commons.lang3.RandomStringUtils;
9   import org.apache.commons.logging.Log;
10  import org.apache.commons.logging.LogFactory;
11  import org.springframework.beans.factory.annotation.Autowired;
12  import org.springframework.boot.autoconfigure.domain.EntityScan;
13  import org.springframework.context.annotation.Bean;
14  import org.springframework.context.annotation.Configuration;
15  import org.springframework.context.annotation.Profile;
16  import org.springframework.core.env.Environment;
17  import org.springframework.data.jpa.repository.config.EnableJpaRepositories;
18  import org.springframework.jdbc.datasource.DriverManagerDataSource;
19  import org.springframework.orm.jpa.JpaTransactionManager;
20  import org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean;
21  import org.springframework.orm.jpa.vendor.HibernateJpaVendorAdapter;
22  import org.springframework.transaction.annotation.EnableTransactionManagement;
23  
24  import java.util.Properties;
25  
26  import javax.persistence.EntityManagerFactory;
27  import javax.sql.DataSource;
28  
29  @Configuration
30  @EnableJpaRepositories(basePackages = {"nl.b3p.tailormap.api.repository"})
31  @EnableTransactionManagement
32  @EntityScan(
33          basePackages = {
34              "nl.tailormap.viewer.config",
35              "nl.tailormap.viewer.config.app",
36              "nl.tailormap.viewer.config.metadata",
37              "nl.tailormap.viewer.config.security",
38              "nl.tailormap.viewer.config.services"
39          })
40  @Profile("test")
41  public class HSQLDBTestProfileJPAConfiguration {
42      private static final Log LOG = LogFactory.getLog(HSQLDBTestProfileJPAConfiguration.class);
43  
44      @Autowired private Environment env;
45  
46      @Bean
47      @Profile("test")
48      public LocalContainerEntityManagerFactoryBean entityManagerFactory() {
49          final LocalContainerEntityManagerFactoryBean em =
50                  new LocalContainerEntityManagerFactoryBean();
51          em.setJpaVendorAdapter(new HibernateJpaVendorAdapter());
52          em.setJpaProperties(additionalProperties());
53          em.setPersistenceUnitName("viewer-config-hsqldb");
54          em.setDataSource(dataSource());
55  
56          return em;
57      }
58  
59      @Bean
60      @Profile("test")
61      public DataSource dataSource() {
62          final DriverManagerDataSource dataSource = new DriverManagerDataSource();
63  
64          String dbUrl = "jdbc:hsqldb:file:./target/unittest-hsqldb-TESTNAMETOKEN/db;shutdown=true";
65          if (null != env.getProperty("spring.datasource.url")) {
66              dbUrl = env.getProperty("spring.datasource.url");
67          }
68          dbUrl = dbUrl.replace("TESTNAMETOKEN", RandomStringUtils.randomAlphabetic(8));
69          dataSource.setUrl(dbUrl);
70          LOG.info("Using test datasource url: " + dataSource.getUrl());
71          dataSource.setDriverClassName("org.hsqldb.jdbcDriver");
72          dataSource.setUsername("sa");
73          dataSource.setPassword("");
74  
75          return dataSource;
76      }
77  
78      @Bean
79      @Profile("test")
80      JpaTransactionManager transactionManager(final EntityManagerFactory entityManagerFactory) {
81          final JpaTransactionManager transactionManager = new JpaTransactionManager();
82          transactionManager.setEntityManagerFactory(entityManagerFactory);
83          return transactionManager;
84      }
85  
86      private Properties additionalProperties() {
87          final Properties hibernateProperties = new Properties();
88  
89          hibernateProperties.setProperty(
90                  "hibernate.hbm2ddl.auto", env.getProperty("spring.jpa.hibernate.ddl-auto"));
91          hibernateProperties.setProperty(
92                  "hibernate.show_sql", env.getProperty("spring.jpa.show-sql", "true"));
93  
94          return hibernateProperties;
95      }
96  }
